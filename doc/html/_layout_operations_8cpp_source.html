40 bool LayoutOperations::performLayoutOp(CorblivarCore& corb, int const& layout_fit_counter, bool const& SA_phase_two, bool const& revertLastOp, bool const& cooling_phase_three) {
41  int op;
42  int die1, tuple1, die2, tuple2, juncts;
43  bool ret;
44 
45  if (LayoutOperations::DBG) {
46  std::cout << "-> LayoutOperations::performLayoutOp(" << &corb << ", " << layout_fit_counter << ", " << SA_phase_two << ", " << revertLastOp << ", " << cooling_phase_three << ")" << std::endl;
47  }
48 
49  // init layout operation variables
50  op = -1;
51  die1 = tuple1 = die2 = tuple2 = juncts = -1;
52 
53  // revert last op
54  if (revertLastOp) {
55  op = this->last_op;
56  }
57  // perform new op
58  else {
59  // special scenario:
60  //
61  // to enable guided block alignment during phase II, we dedicatedly handle
62  // particular blocks of failing requests / the requests themselves;
63  // however, this should only be considered for cooling phase 3, i.e., when
64  // some local minima is reached; otherwise, when these operations are
65  // applied too often, the cost function will largely vary in value and is
66  // thus not an appropriate measure for guided minimization anymore
67  //
68  // note that the related handlers below do only return true in case any
69  // failed alignment exists
70  if (SA_phase_two && this->parameters.opt_alignment && cooling_phase_three) {
71 
72  // randomly decide to try either block swapping or swapping
73  // coordinates of flexible alignments
74  //
75  if (Math::randB()) {
76 
77  // try to setup swapping failed blocks
78  if (this->prepareBlockSwappingFailedAlignment(corb, die1, tuple1, die2, tuple2)) {
79 
80  // engage swap operation; this dedicated _ENFORCE
81  // op-code ignores power-aware block-die
82  // assignments, such that block alignments can be
83  // also fulfilled when the option power-aware
84  // block assignment is activated
85  this->last_op = op = LayoutOperations::OP_SWAP_BLOCKS_ENFORCE;
86  }
87  }
88  // other order of trying operations
89  else {
90 
91  // try to setup swapping alignment coordinates
92  if (this->prepareSwappingCoordinatesFailedAlignment(corb, tuple1)) {
93 
94  this->last_op = op = LayoutOperations::OP_SWAP_ALIGNMENT_COORDINATES;
95  }
96  }
97  }
98  // another scenario, in case no valid layout was found at all in the
99  // previous iteration: randomly select one block which currently exceeds
100  // the fixed outline
101  else if (!SA_phase_two && layout_fit_counter == 0) {
102 
103  if (Math::randB()) {
104  this->prepareHandlingOutlineCriticalBlock(corb, die1, tuple1);
105 
106  // perform any random
107  this->last_op = op = Math::randI(1, 6);
108  }
109  }
110  // in case no valid layout was found recently (but previously already a
111  // valid layout was found), randomly select to reshape a block exceeding
112  // the outline
113  else if (SA_phase_two && layout_fit_counter == 0) {
114 
115  if (Math::randB()) {
116  this->prepareHandlingOutlineCriticalBlock(corb, die1, tuple1);
117  this->last_op = op = LayoutOperations::OP_ROTATE_BLOCK__SHAPE_BLOCK;
118  }
119  }
120 
121  // for other (regular) cases or if special scenarios above cannot be
122  // performed or if they still require random operations, define a random
123  // operation next
124  //
125  if (op == -1) {
126 
127  // randomly select one block from currently largest net w/ highest
128  // individual impact on WL
129  if (Math::randB()) {
130  this->preselectBlockFromLargestNet(corb, die1, tuple1);
131  }
132 
133  // see defined op-codes to set random-number ranges; recall that
134  // randI(x,y) is [x,y)
135  this->last_op = op = Math::randI(1, 6);
136  }
137  }
138 
139  // specific op handler
140  switch (op) {
141 
142  case LayoutOperations::OP_SWAP_BLOCKS: // op-code: 1
143 
144  ret = this->performOpMoveOrSwapBlocks(LayoutOperations::OP_SWAP_BLOCKS, revertLastOp, !SA_phase_two, corb, die1, die2, tuple1, tuple2);
145 
146  break;
147 
148  case LayoutOperations::OP_MOVE_TUPLE: // op-code: 2
149 
150  ret = this->performOpMoveOrSwapBlocks(LayoutOperations::OP_MOVE_TUPLE, revertLastOp, !SA_phase_two, corb, die1, die2, tuple1, tuple2);
151 
152  break;
153 
154  case LayoutOperations::OP_SWITCH_INSERTION_DIR: // op-code: 3
155 
156  ret = this->performOpSwitchInsertionDirection(revertLastOp, corb, die1, tuple1);
157 
158  break;
159 
160  case LayoutOperations::OP_SWITCH_TUPLE_JUNCTS: // op-code: 4
161 
162  ret = this->performOpSwitchTupleJunctions(revertLastOp, corb, die1, tuple1, juncts);
163 
164  break;
165 
166  case LayoutOperations::OP_ROTATE_BLOCK__SHAPE_BLOCK: // op-code: 5
167 
168  ret = this->performOpShapeBlock(revertLastOp, corb, die1, tuple1);
169 
170  break;
171 
172  case LayoutOperations::OP_SWAP_BLOCKS_ENFORCE: // op-code: 20
173 
174  ret = this->performOpMoveOrSwapBlocks(LayoutOperations::OP_SWAP_BLOCKS_ENFORCE, revertLastOp, !SA_phase_two, corb, die1, die2, tuple1, tuple2);
175 
176  break;
177 
178  case LayoutOperations::OP_SWAP_ALIGNMENT_COORDINATES: // op-code: 21
179 
180  ret = this->performOpSwapAlignmentCoordinates(revertLastOp, corb, tuple1);
181 
182  break;
183  }
184 
185  // memorize elements of successful op
186  if (ret) {
187  this->last_op_die1 = die1;
188  this->last_op_die2 = die2;
189  this->last_op_tuple1 = tuple1;
190  this->last_op_tuple2 = tuple2;
191  this->last_op_juncts = juncts;
192  }
193 
194  if (LayoutOperations::DBG) {
195  std::cout << "<- LayoutOperations::performLayoutOp : " << ret << std::endl;
196  }
197 
198  return ret;
199 }

202  int random_tuple;
203 
204  tuple1 = -1;
205 
206  // randomly decide whether to work on the x- or y-dimension; this part is
207  // for x-direction
208  if (Math::randB()) {
209 
210  // search for one critical block among all dies
211  for (int l = 0; l < this->parameters.layers; l++) {
212 
213  for (unsigned b = 0; b < corb.getDie(l).getBlocks().size(); b++) {
214 
215  // randomly consider any block on the current die;
216  // when it's exceeding the outline it's to be
217  // altered

219 
220  // current block exceeding die width?

222  die1 = l;
223  tuple1 = random_tuple;
224  break;
225  }
226  }
227 
228  if (tuple1 != -1) {
229  break;
230  }
231  }
232  }
233  // randomly decide whether to work on the x- or y-dimension; this part is for
234  // y-direction
235  else {
236  // search for one critical block among all dies
237  for (int l = 0; l < this->parameters.layers; l++) {
238 
239  for (unsigned b = 0; b < corb.getDie(l).getBlocks().size(); b++) {
240

286 bool LayoutOperations::prepareBlockSwappingFailedAlignment(CorblivarCore const& corb, int& die1, int& tuple1, int& die2, int& tuple2) {
287  CorblivarAlignmentReq const* failed_req = nullptr;
288  std::vector<CorblivarAlignmentReq const*> failed_reqs;
289  Block const* b1;
290  Block const* b1_partner;
291  Block const* b1_neighbour = nullptr;
292  Rect bb;
293 
294  // determine failed alignments
295  for (unsigned r = 0; r < corb.getAlignments().size(); r++) {
296 
297  if (!corb.getAlignments()[r].fulfilled) {
298  failed_reqs.push_back(&corb.getAlignments()[r]);
299  }
300  }
301 
302  // randomly pick any failed alignment
303  if (!failed_reqs.empty()) {
304 
305  failed_req = failed_reqs[Math::randI(0, failed_reqs.size())];
306 
307  // randomly decide for one block to move around / to swap with other
308  // blocks; avoid the dummy reference block if required
309  if (
310  // randomly select s_i if it's not the RBOD
311  (failed_req->s_i->numerical_id != RBOD::NUMERICAL_ID && Math::randB()) ||
312  // also consider s_i if s_j is the RBOD
313  failed_req->s_j->numerical_id == RBOD::NUMERICAL_ID
314  ) {
315  // sanity check for both s_i and s_j being RBOD
316  if (failed_req->s_i->numerical_id == RBOD::NUMERICAL_ID) {
317  return false;
318  }
319 
320  // s_i is the block to be changed
321  b1 = failed_req->s_i;
322  // memorize the opposite block s_j
323  b1_partner = failed_req->s_j;
324  }
325  else {
326  // s_j is the block to be changed
327  b1 = failed_req->s_j;
328  // memorize the opposite block s_i
329  b1_partner = failed_req->s_i;
330  }
331 
332  // determine die and tuple variables; tuple2 is to be determined below
333  //
334  die1 = b1->layer;
335  tuple1 = corb.getDie(die1).getTuple(b1);
336  // for RBOD being the partner, we assume the same die as for the block to
337  // be changed
338  if (b1_partner->numerical_id == RBOD::NUMERICAL_ID) {
339  die2 = die1;
340  }
341  else {
342  die2 = b1_partner->layer;
343  }
344 
345  if (CorblivarAlignmentReq::DBG_HANDLE_FAILED) {
346  std::cout << "DBG_ALIGNMENT> s_i: " << failed_req->s_i->id << std::endl;
347  std::cout << "DBG_ALIGNMENT> s_j: " << failed_req->s_j->id << std::endl;
348  std::cout << "DBG_ALIGNMENT> b1: " << b1->id << std::endl;
349  std::cout << "DBG_ALIGNMENT> b1_partner: " << b1_partner->id << std::endl;
350  std::cout << "DBG_ALIGNMENT> die1: " << die1 << std::endl;
351  std::cout << "DBG_ALIGNMENT> tuple1: " << tuple1 << std::endl;
352  std::cout << "DBG_ALIGNMENT> die2: " << die2 << std::endl;
353  std::cout << "DBG_ALIGNMENT> tuple1: to be determined" << std::endl;
354  }
355 
356  // dedicatedly defined vertical bus; failed vertical alignment across
357  // different dies
358  if (failed_req->vertical_bus()) {
359 
360  // select block to swap with b1 such that blocks to be aligned (b1
361  // and its partner) are initially at least intersecting blocks;
362  // that means, we need to swap with a block that is intersecting
363  // with b1's partner block
364  //
365  // if b1 and its partner block are in one die, b1 needs to be
366  // swapped with a block on another layer which is intersecting
367  // b1's partner block; randomly select another layer
368  //
369  if (die1 == die2) {
370 
371  // such vertical alignment is only possible for > 1
372  // layers; sanity check
373  if (this->parameters.layers == 1) {
374  return false;
375  }
376 
377  while (die1 == die2) {
378  die2 = Math::randI(0, this->parameters.layers);
379  }
380  }
381  // if b1 and its partner block are in different dies, b1 can be
382  // swapped with a block intersecting b1's partner block on the
383  // current die of b1
384  else {
385  die2 = die1;
386  }
387 
388  // the block to swap w/ is stepwise searched according to this bb;
389  // init it with the bb of b1's partner block
390  bb = b1_partner->bb;
391 
392  while (true) {
393 
394  for (Block const* b2 : corb.getDie(die2).getBlocks()) {
395 
396  // candidate block; overlaps with b1's partner
397  // block
398  if (Rect::rectsIntersect(bb, b2->bb) &&
399  // avoid swapping with b1 itself
400  b1->numerical_id != b2->numerical_id &&
401  // also check that blocks are not partner blocks
402  // of the alignment request; otherwise,
403  // consecutively circular swap might occur which
404  // are not resolving the failing alignment
405  !failed_req->partner_blocks(b1, b2)
406  ) {
407 
408  b1_neighbour = b2;
409 
410  break;
411  }
412  }
413 
414  // no intersecting block was found, increase the search
415  // radius by doubling the considered bb
416  if (b1_neighbour == nullptr) {
417 
418  bb.ll.x -= bb.w / 2.0;
419  bb.ur.x += bb.w / 2.0;
420  bb.ll.y -= bb.h / 2.0;
421  bb.ur.y += bb.h / 2.0;
422  }
423  else {
424  break;
425  }
426  }
427 
428  }
429 
430  // other failed alignment ranges or non-zero-offset fixed alignment
431  //
432  // determine relevant neighbour block to perform swap operation, i.e.,
433  // nearest neighbour w.r.t. failure type
434  else {
435 
436  // also consider to randomly change die2 as well; this is required
437  // for alignments which cannot be fulfilled within one die and
438  // does not harm for alignments which could be fulfilled within
439  // one die (they can then also be fulfilled across dies); note
440  // that an explicit check for all the different options of
441  // alignments not possible within one die are not performed here
442  // but rather a die change is considered randomly
443  //
444  // note that changing dies is only possible for > 1 layers
445  if (Math::randB() && this->parameters.layers > 1) {
446 
447  while (die1 == die2) {
448  die2 = Math::randI(0, this->parameters.layers);
449  }
450  }
451 
452  // consider different neighbours for different alignment failures
453  switch (b1->alignment) {
454 
455  // determine nearest right block
456  case Block::AlignmentStatus::FAIL_HOR_TOO_LEFT:
457 
458  for (Block const* b2 : corb.getDie(die2).getBlocks()) {
459 
460  if (Rect::rectA_leftOf_rectB(b1->bb, b2->bb, true) &&
461  // also check that blocks are not partner blocks
462  // of the alignment request; otherwise,
463  // consecutively circular swap might occur which
464  // are not resolving the failing alignment
465  !failed_req->partner_blocks(b1, b2)
466  ) {
467 
468  if (b1_neighbour == nullptr || b2->bb.ll.x < b1_neighbour->bb.ll.x) {
469  b1_neighbour = b2;
470  }
471  }
472  }
473 
474  break;
475 
476  // determine nearest left block
477  case Block::AlignmentStatus::FAIL_HOR_TOO_RIGHT:
478 
479  for (Block const* b2 : corb.getDie(die2).getBlocks()) {
480 
481  if (Rect::rectA_leftOf_rectB(b2->bb, b1->bb, true) &&
482  // also check that blocks are not partner blocks
483  // of the alignment request; otherwise,
484  // consecutively circular swap might occur which
485  // are not resolving the failing alignment
486  !failed_req->partner_blocks(b1, b2)
487  ) {
488 
489  if (b1_neighbour == nullptr || b2->bb.ur.x > b1_neighbour->bb.ur.x) {
490  b1_neighbour = b2;
491  }
492  }
493  }
494 
495  break;
496 
497  // determine nearest block above
498  case Block::AlignmentStatus::FAIL_VERT_TOO_LOW:
499 
500  for (Block const* b2 : corb.getDie(die2).getBlocks()) {
501 
502  if (Rect::rectA_below_rectB(b1->bb, b2->bb, true) &&
503  // also check that blocks are not partner blocks
504  // of the alignment request; otherwise,
505  // consecutively circular swap might occur which
506  // are not resolving the failing alignment
507  !failed_req->partner_blocks(b1, b2)
508  ) {
509 
510  if (b1_neighbour == nullptr || b2->bb.ll.y < b1_neighbour->bb.ll.y) {
511  b1_neighbour = b2;
512  }
513  }
514  }
515 
516  break;
517 
518  // determine nearest block below
519  case Block::AlignmentStatus::FAIL_VERT_TOO_HIGH:
520 
521  for (Block const* b2 : corb.getDie(die2).getBlocks()) {
522 
523  if (Rect::rectA_below_rectB(b2->bb, b1->bb, true) &&
524  // also check that blocks are not partner blocks
525  // of the alignment request; otherwise,
526  // consecutively circular swap might occur which
527  // are not resolving the failing alignment
528  !failed_req->partner_blocks(b1, b2)
529  ) {
530 
531  if (b1_neighbour == nullptr || b2->bb.ur.y > b1_neighbour->bb.ur.y) {
532  b1_neighbour = b2;
533  }
534  }
535  }
536 
537  break;
538 
539  // dummy case, to catch other (here not occurring)
540  // alignment status
541  default:
542  break;
543  }
544  }
545 
546  // determine related tuple of neighbour block; == -1 in case the tuple
547  // cannot be find; sanity check for undefined neighbour
548  if (b1_neighbour != nullptr) {
549 
550  tuple2 = corb.getDie(die2).getTuple(b1_neighbour);
551 
552  if (CorblivarAlignmentReq::DBG_HANDLE_FAILED) {
553  std::cout << "DBG_ALIGNMENT> " << failed_req->tupleString() << " failed so far;" << std::endl;
554  std::cout << "DBG_ALIGNMENT> failed alignment status (" << b1->id << "): " << b1->alignment << std::endl;
555  std::cout << "DBG_ALIGNMENT> considering swapping block " << b1->id << " on layer " << b1->layer;
556  std::cout << " with block " << b1_neighbour->id << " on layer " << b1_neighbour->layer << std::endl;
557  }
558 
559  return true;
560  }
561  else {
562  tuple2 = -1;
563 
564  if (CorblivarAlignmentReq::DBG_HANDLE_FAILED) {
565  std::cout << "DBG_ALIGNMENT> " << failed_req->tupleString() << " failed so far;" << std::endl;
566  std::cout << "DBG_ALIGNMENT> failed alignment status (" << b1->id << "): " << b1->alignment << std::endl;
567  std::cout << "DBG_ALIGNMENT> no appropriate block to swap with found" << std::endl;
568  }
569 
570  return false;
571  }
572  }
573 
574  // no failing request was found
575  else {
576  return false;
577  }
578 }

580 bool LayoutOperations::performOpEnhancedSoftBlockShaping(CorblivarCore const& corb, Block const* shape_block) const {
581  int op;
582  double boundary_x, boundary_y;
583  double width, height;
584 
585  // see defined op-codes in class FloorPlanner to set random-number ranges;
586  // recall that randI(x,y) is [x,y)
587  op = Math::randI(10, 15);
588 
589  switch (op) {
590 
591  // stretch such that shape_block's right front aligns w/ the right front
592  // of the nearest other block
593  case LayoutOperations::OP_SHAPE_BLOCK__STRETCH_HORIZONTAL: // op-code: 10
594 
595  // dummy value, to be large than right front
596  boundary_x = 2.0 * shape_block->bb.ur.x;
597 
598  for (Block const* b : corb.getDie(shape_block->layer).getBlocks()) {
599 
600  // determine nearest right front of other blocks
601  if (b->bb.ur.x > shape_block->bb.ur.x) {
602  boundary_x = std::min(boundary_x, b->bb.ur.x);
603  }
604  }
605 
606  // determine resulting new dimensions
607  width = boundary_x - shape_block->bb.ll.x;
608  height = shape_block->bb.area / width;
609 
610  // apply new dimensions in case the resulting AR is allowed
611  return shape_block->shapeByWidthHeight(width, height);
612 
613  // shrink such that shape_block's right front aligns w/ the left front of
614  // the nearest other block
615  case LayoutOperations::OP_SHAPE_BLOCK__SHRINK_HORIZONTAL: // op-code: 12
616 
617  boundary_x = 0.0;
618 
619  for (Block const* b : corb.getDie(shape_block->layer).getBlocks()) {
620 
621  // determine nearest left front of other blocks
622  if (b->bb.ll.x < shape_block->bb.ur.x) {
623  boundary_x = std::max(boundary_x, b->bb.ll.x);
624  }
625  }
626 
627  // determine resulting new dimensions
628  width = boundary_x - shape_block->bb.ll.x;
629  height = shape_block->bb.area / width;
630 
631  // apply new dimensions in case the resulting AR is allowed
632  return shape_block->shapeByWidthHeight(width, height);
633 
634  // stretch such that shape_block's top front aligns w/ the top front of
635  // the nearest other block
636  case LayoutOperations::OP_SHAPE_BLOCK__STRETCH_VERTICAL: // op-code: 11
637 
638  // dummy value, to be large than top front
639  boundary_y = 2.0 * shape_block->bb.ur.y;
640 
641  for (Block const* b : corb.getDie(shape_block->layer).getBlocks()) {
642 
643  // determine nearest top front of other blocks
644  if (b->bb.ur.y > shape_block->bb.ur.y) {
645  boundary_y = std::min(boundary_y, b->bb.ur.y);
646  }
647  }
648 
649  // determine resulting new dimensions
650  height = boundary_y - shape_block->bb.ll.y;
651  width = shape_block->bb.area / height;
