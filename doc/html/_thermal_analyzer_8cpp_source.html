158 void ThermalAnalyzer::initPowerMaps(int const& layers, Point const& die_outline) {
159  unsigned b;
160  int i;
161  ThermalAnalyzer::PowerMapBin init_bin;
162 
163  if (ThermalAnalyzer::DBG_CALLS) {
164  std::cout << "-> ThermalAnalyzer::initPowerMaps(" << layers << ", " << die_outline.x << ", " << die_outline.y << ")" << std::endl;
165  }
166 
167  this->power_maps.clear();
168 
169  // allocate power-maps arrays
170  for (i = 0; i < layers; i++) {
171  this->power_maps.emplace_back(
172  std::array<std::array<ThermalAnalyzer::PowerMapBin, ThermalAnalyzer::POWER_MAPS_DIM>, ThermalAnalyzer::POWER_MAPS_DIM>()
173  );
174  }
175 
176  // init the maps w/ zero values
177  init_bin.power_density = init_bin.TSV_density = 0.0;
178  for (i = 0; i < layers; i++) {
179  for (auto& partial_map : this->power_maps[i]) {
180  partial_map.fill(init_bin);
181  }
182  }
183 
184  // scale power map dimensions to outline of thermal map; this way the padding of
185  // power maps doesn't distort the block outlines in the thermal map
186  this->power_maps_dim_x = die_outline.x / ThermalAnalyzer::THERMAL_MAP_DIM;
187  this->power_maps_dim_y = die_outline.y / ThermalAnalyzer::THERMAL_MAP_DIM;
188 
189  // determine offset for blocks, related to padding of power maps
190  this->blocks_offset_x = this->power_maps_dim_x * ThermalAnalyzer::POWER_MAPS_PADDED_BINS;
191  this->blocks_offset_y = this->power_maps_dim_y * ThermalAnalyzer::POWER_MAPS_PADDED_BINS;
192 
193  // determine max distance for blocks' upper/right boundaries to upper/right die
194  // outline to be padded
195  this->padding_right_boundary_blocks_distance = ThermalAnalyzer::PADDING_ZONE_BLOCKS_DISTANCE_LIMIT * die_outline.x;
196  this->padding_upper_boundary_blocks_distance = ThermalAnalyzer::PADDING_ZONE_BLOCKS_DISTANCE_LIMIT * die_outline.y;
197 
198  // predetermine map bins' area and lower-left corner coordinates; note that the
199  // last bin represents the upper-right coordinates for the penultimate bin
200  this->power_maps_bin_area = this->power_maps_dim_x * this->power_maps_dim_y;
201  for (b = 0; b <= ThermalAnalyzer::POWER_MAPS_DIM; b++) {
202  this->power_maps_bins_ll_x[b] = b * this->power_maps_dim_x;
203  }
204  for (b = 0; b <= ThermalAnalyzer::POWER_MAPS_DIM; b++) {
205  this->power_maps_bins_ll_y[b] = b * this->power_maps_dim_y;
206  }
207 
208  if (ThermalAnalyzer::DBG_CALLS) {
209  std::cout << "<- ThermalAnalyzer::initPowerMaps" << std::endl;
210  }

219 void ThermalAnalyzer::initThermalMasks(int const& layers, bool const& log, MaskParameters const& parameters) {
220  int i, ii;
221  double scale;
222  double layer_impulse_factor;
223  int x_y;
224 
225  if (ThermalAnalyzer::DBG_CALLS) {
226  std::cout << "-> ThermalAnalyzer::initThermalMasks(" << layers << ", " << log << ")" << std::endl;
227  }
228 
229  if (log) {
230  std::cout << "ThermalAnalyzer> ";
231  std::cout << "Initializing thermals masks for power blurring ..." << std::endl;
232  }
233 
234  // reset mask arrays
235  this->thermal_masks.clear();
236 
237  // allocate mask arrays
238  for (i = 0; i < layers; i++) {
239  this->thermal_masks.emplace_back(
240  std::array<double,ThermalAnalyzer::THERMAL_MASK_DIM>()
241  );
242  }
243 
244  // determine scale factor such that mask_boundary_value is reached at the
245  // boundary of the lowermost (2D) mask; based on general 2D gauss equation,
246  // determines gauss(x = y) = mask_boundary_value;
247  // constant spread (e.g., 1.0) is sufficient since this function fitting
248  // only requires two parameters, i.e., varying spread has no impact
249  static constexpr double SPREAD = 1.0;
250  // scaling is required for function fitting; the maximum of the gauss / exp
251  // function is defined by the impulse factor, the minimum by the
252  // mask_boundary_value
253  scale = std::sqrt(SPREAD * std::log(parameters.impulse_factor / (parameters.mask_boundary_value))) / std::sqrt(2.0);
254  // normalize factor according to half of mask dimension; i.e., fit spreading of
255  // exp function
256  scale /= ThermalAnalyzer::THERMAL_MASK_CENTER;
257 
258  // determine all masks, starting from lowest layer, i.e., hottest layer
259  for (i = 1; i <= layers; i++) {
260 
261  // impulse factor is to be reduced notably for increasing layer count
262  layer_impulse_factor = parameters.impulse_factor / std::pow(i, parameters.impulse_factor_scaling_exponent);
263 
264  ii = 0;
265  for (x_y = -ThermalAnalyzer::THERMAL_MASK_CENTER; x_y <= ThermalAnalyzer::THERMAL_MASK_CENTER; x_y++) {
266  // sqrt for impulse factor is mandatory since the mask is
267  // used for separated convolution (i.e., factor will be
268  // squared in final convolution result)
269  this->thermal_masks[i - 1][ii] = Math::gauss1D(x_y * scale, std::sqrt(layer_impulse_factor), SPREAD);
270 
271  ii++;
272  }
273  }
274 
275  if (ThermalAnalyzer::DBG) {
276  // enforce fixed digit count for printing mask
277  std::cout << std::fixed;
278  // dump mask
279  for (i = 0; i < layers; i++) {
280  std::cout << "DBG> Thermal 1D mask for point source on layer " << i << ":" << std::endl;
281  for (x_y = 0; x_y < ThermalAnalyzer::THERMAL_MASK_DIM; x_y++) {
282  std::cout << this->thermal_masks[i][x_y] << ", ";
283  }
284  std::cout << std::endl;
285  }
286  // reset to default floating output
287  std::cout.unsetf(std::ios_base::floatfield);
288 
289  std::cout << std::endl;
290  std::cout << "DBG> Note that these values will be multiplied w/ each other in the final 2D mask" << std::endl;
291  }
292 
293  if (log) {
294  std::cout << "ThermalAnalyzer> ";
295  std::cout << "Done" << std::endl << std::endl;
296  }
297 
298  if (ThermalAnalyzer::DBG_CALLS) {
299  std::cout << "<- ThermalAnalyzer::initThermalMasks" << std::endl;
300  }
301 }

303 void ThermalAnalyzer::generatePowerMaps(int const& layers, std::vector<Block> const& blocks, Point const& die_outline, MaskParameters const& parameters, bool const& extend_boundary_blocks_into_padding_zone) {
304  int i;
305  int x, y;
306  Rect bin, intersect, block_offset;
307  int x_lower, x_upper, y_lower, y_upper;
308  bool padding_zone;
309  ThermalAnalyzer::PowerMapBin init_bin;
310 
311  if (ThermalAnalyzer::DBG_CALLS) {
312  std::cout << "-> ThermalAnalyzer::generatePowerMaps(" << layers << ", " << &blocks << ", (" << die_outline.x << ", " << die_outline.y << "), " << &parameters << ", " << extend_boundary_blocks_into_padding_zone << ")" << std::endl;
313  }
314 
315  init_bin.power_density = 0.0;
316  init_bin.TSV_density = 0.0;
317 
318  // determine maps for each layer
319  for (i = 0; i < layers; i++) {
320 
321  // reset map to zero
322  // note: this also implicitly pads the map w/ zero power density
323  for (auto& m : this->power_maps[i]) {
324  m.fill(init_bin);
325  }
326 
327  // consider each block on the related layer
328  for (Block const& block : blocks) {
329 
330  // drop blocks assigned to other layers
331  if (block.layer != i) {
332  continue;
333  }
334 
335  // determine offset, i.e., shifted, block bb; relates to block's
336  // bb in padded power map
337  block_offset = block.bb;
338 
339  // don't offset blocks at the left/lower chip boundaries,
340  // implicitly extend them into power-map padding zone; this way,
341  // during convolution, the thermal estimate increases for these
342  // blocks; blocks not at the boundaries are shifted
343  if (extend_boundary_blocks_into_padding_zone && block.bb.ll.x == 0.0) {
344  }
345  else {
346  block_offset.ll.x += this->blocks_offset_x;
347  }
348  if (extend_boundary_blocks_into_padding_zone && block.bb.ll.y == 0.0) {
349  }
350  else {
351  block_offset.ll.y += this->blocks_offset_y;
352  }
353 
354  // also consider extending blocks into right/upper padding zone if
355  // they are close to the related chip boundaries
356  if (
357  extend_boundary_blocks_into_padding_zone &&
358  std::abs(die_outline.x - block.bb.ur.x) < this->padding_right_boundary_blocks_distance
359  ) {
360  // consider offset twice in order to reach right/uppper
361  // boundary related to layout described by padded power map
362  block_offset.ur.x = die_outline.x + 2.0 * this->blocks_offset_x;
363  }
364  // simple shift otherwise; compensate for padding of left/bottom
365  // boundaries
366  else {
367  block_offset.ur.x += this->blocks_offset_x;
368  }
369 
370  if (
371  extend_boundary_blocks_into_padding_zone
372  && std::abs(die_outline.y - block.bb.ur.y) < this->padding_upper_boundary_blocks_distance
373  ) {
374  block_offset.ur.y = die_outline.y + 2.0 * this->blocks_offset_y;
375  }
376  else {
377  block_offset.ur.y += this->blocks_offset_y;
378  }
379 
380  // determine index boundaries for offset block; based on boundary
381  // of blocks and the covered bins; note that cast to int truncates
382  // toward zero, i.e., performs like floor for positive numbers
383  x_lower = static_cast<int>(block_offset.ll.x / this->power_maps_dim_x);
384  y_lower = static_cast<int>(block_offset.ll.y / this->power_maps_dim_y);
385  // +1 in order to efficiently emulate the result of ceil(); limit
386  // upper bound to power-maps dimenions
387  x_upper = std::min(static_cast<int>(block_offset.ur.x / this->power_maps_dim_x) + 1, ThermalAnalyzer::POWER_MAPS_DIM);
388  y_upper = std::min(static_cast<int>(block_offset.ur.y / this->power_maps_dim_y) + 1, ThermalAnalyzer::POWER_MAPS_DIM);
389 
390  // walk power-map bins covering block outline
391  for (x = x_lower; x < x_upper; x++) {
392  for (y = y_lower; y < y_upper; y++) {
393 
394  // determine if bin w/in padding zone
395  if (
396  x < ThermalAnalyzer::POWER_MAPS_PADDED_BINS
397  || x >= (ThermalAnalyzer::POWER_MAPS_DIM - ThermalAnalyzer::POWER_MAPS_PADDED_BINS)
398  || y < ThermalAnalyzer::POWER_MAPS_PADDED_BINS
399  || y >= (ThermalAnalyzer::POWER_MAPS_DIM - ThermalAnalyzer::POWER_MAPS_PADDED_BINS)
400  ) {
401  padding_zone = true;
402  }
403  else {
404  padding_zone = false;
405  }
406 
407  // consider full block power density for fully covered bins
408  if (x_lower < x && x < (x_upper - 1) && y_lower < y && y < (y_upper - 1)) {
409  if (padding_zone) {
410  this->power_maps[i][x][y].power_density += block.power_density() * parameters.power_density_scaling_padding_zone;
411  }
412  else {
413  this->power_maps[i][x][y].power_density += block.power_density();
414  }
415  }
416  // else consider block power according to
417  // intersection of current bin and block
418  else {
419  // determine real coords of map bin
420  bin.ll.x = this->power_maps_bins_ll_x[x];
421  bin.ll.y = this->power_maps_bins_ll_y[y];
422  // note that +1 is guaranteed to be within bounds
423  // of power_maps_bins_ll_x/y (size =
424  // ThermalAnalyzer::POWER_MAPS_DIM + 1); the
425  // related last tuple describes the upper-right
426  // corner coordinates of the right/top boundary
427  bin.ur.x = this->power_maps_bins_ll_x[x + 1];
428  bin.ur.y = this->power_maps_bins_ll_y[y + 1];
429 
430  // determine intersection
431  intersect = Rect::determineIntersection(bin, block_offset);
432  // normalize to full bin area
433  intersect.area /= this->power_maps_bin_area;
434 
435  if (padding_zone) {
436  this->power_maps[i][x][y].power_density += block.power_density() * intersect.area * parameters.power_density_scaling_padding_zone;
437  }
438  else {
439  this->power_maps[i][x][y].power_density += block.power_density() * intersect.area;
440  }
441  }
442  }
443  }
444  }
445  }
446 
447  if (ThermalAnalyzer::DBG_CALLS) {
448  std::cout << "<- ThermalAnalyzer::generatePowerMaps" << std::endl;
449  }
450 }

458  Rect bin, bin_intersect;
459  int x_lower, x_upper, y_lower, y_upper;

486  // bound to power-maps dimensions

488  y_upper = std::min(static_cast<int>(TSV_group.bb.ur.y / this->power_maps_dim_y) + 1, ThermalAnalyzer::POWER_MAPS_DIM);
489 
490  if (ThermalAnalyzer::DBG) {
491  std::cout << "DBG> TSV group " << TSV_group.id << std::endl;
492  std::cout << "DBG> Affected power-map bins: " << x_lower << "," << y_lower
493  << " to " <<
494  x_upper << "," << y_upper << std::endl;
495  }
496 
497  // walk power-map bins covering intersection outline; adapt TSV densities
498  for (x = x_lower; x < x_upper; x++) {
499  for (y = y_lower; y < y_upper; y++) {
500 
501  // consider full TSV density for fully covered bins
502  if (x_lower < x && x < (x_upper - 1) && y_lower < y && y < (y_upper - 1)) {
503 
504  // adapt map on affected layer
505  this->power_maps[TSV_group.layer][x][y].TSV_density += 100.0;
506  }
507  // else consider TSV density according to partial
508  // intersection with current bin
509  else {
510  // determine real coords of map bin
511  bin.ll.x = this->power_maps_bins_ll_x[x];
512  bin.ll.y = this->power_maps_bins_ll_y[y];
513  // note that +1 is guaranteed to be within
514  // bounds of power_maps_bins_ll_x/y (size
515  // = ThermalAnalyzer::POWER_MAPS_DIM + 1);
516  // the related last tuple describes the
517  // upper-right corner coordinates of the
518  // right/top boundary
519  bin.ur.x = this->power_maps_bins_ll_x[x + 1];
520  bin.ur.y = this->power_maps_bins_ll_y[y + 1];
521 
522  // determine intersection
523  bin_intersect = Rect::determineIntersection(bin, TSV_group.bb);
524  // normalize to full bin area
525  bin_intersect.area /= this->power_maps_bin_area;
526 
527  // adapt map on affected layer
528  this->power_maps[TSV_group.layer][x][y].TSV_density += 100.0 * bin_intersect.area;
529  }
530  }
531  }
532  }
533 
534  // walk power-map bins; adapt power according to TSV densities
535  for (x = ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x < ThermalAnalyzer::THERMAL_MAP_DIM + ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x++) {
536  for (y = ThermalAnalyzer::POWER_MAPS_PADDED_BINS; y < ThermalAnalyzer::THERMAL_MAP_DIM + ThermalAnalyzer::POWER_MAPS_PADDED_BINS; y++) {
537 
538  // sanity check; TSV density should be <= 100%; might be larger
539  // due to superposition in calculations above
540  for (i = 0; i < layers; i++) {
541  this->power_maps[i][x][y].TSV_density = std::min(100.0, this->power_maps[i][x][y].TSV_density);
542  }
543 
544  // adapt maps for all layers; the uppermost layer next the
545  // heatsink may also contain (dummy) thermal TSVs
546  for (i = 0; i < layers; i++) {
547 
548  // ignore cases w/o TSVs
549  if (this->power_maps[i][x][y].TSV_density == 0.0)
550  continue;
551 
552  // scaling depends on TSV density; the larger the TSV
553  // density, the larger the power down-scaling; note that
554  // the factor power_density_scaling_TSV_region ranges b/w
555  // 0.0 and 1.0, whereas TSV_density ranges b/w 0.0 and
556  // 100.0
557  this->power_maps[i][x][y].power_density *=
558  1.0 +
559  ((parameters.power_density_scaling_TSV_region - 1.0) / 100.0) * this->power_maps[i][x][y].TSV_density;
560  }
561  }
562  }
563 
564  if (ThermalAnalyzer::DBG_CALLS) {
565  std::cout << "<- ThermalAnalyzer::adaptPowerMapsTSVs" << std::endl;
566  }

624 void ThermalAnalyzer::performPowerBlurring(ThermalAnalysisResult& ret, int const& layers, MaskParameters const& parameters) {
625  int layer;
626  int x, y, i;
627  int map_x, map_y;
628  int mask_i;
629  double max_temp, avg_temp;
630  // required as buffer for separated convolution; note that its dimensions
631  // corresponds to a power map, which is required to hold temporary results for 1D
632  // convolution of padded power maps
633  std::array< std::array<double, ThermalAnalyzer::POWER_MAPS_DIM>, ThermalAnalyzer::POWER_MAPS_DIM> thermal_map_tmp;
634 
635  if (ThermalAnalyzer::DBG_CALLS) {
636  std::cout << "-> ThermalAnalyzer::performPowerBlurring(" << &ret << ", " << ", " << layers << ", " << &parameters << ")" << std::endl;
637  }
638 
639  // init temp map w/ zero
640  for (auto& m : thermal_map_tmp) {
641  m.fill(0.0);
642  }
643 
644  // init final map w/ temperature offset; offset is a additive factor, and thus not
645  // considered during convolution
646  for (x = 0; x < ThermalAnalyzer::THERMAL_MAP_DIM; x++) {
647  for (y = 0; y < ThermalAnalyzer::THERMAL_MAP_DIM; y++) {
648  this->thermal_map[x][y].temp = parameters.temp_offset;
649  }
650  }
651 
654  //
655  // start w/ horizontal convolution (with which to start doesn't matter actually)
656  for (layer = 0; layer < layers; layer++) {
657 
658  // walk power-map grid for horizontal convolution; store into
659  // thermal_map_tmp; note that during horizontal convolution we need to
660  // walk the full y-dimension related to the padded power map in order to
661  // reasonably model the thermal effect in the padding zone during
662  // subsequent vertical convolution
663  for (y = 0; y < ThermalAnalyzer::POWER_MAPS_DIM; y++) {
664 
665  // for the x-dimension during horizontal convolution, we need to
666  // restrict the considered range according to the thermal map in
667  // order to exploit the padded power map w/o mask boundary checks
668  for (x = ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x < ThermalAnalyzer::THERMAL_MAP_DIM + ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x++) {
669 
670  // perform horizontal 1D convolution, i.e., multiply
671  // input[x] w/ mask
672  //
673  // e.g., for x = 0, THERMAL_MASK_DIM = 3
674  // convol1D(x=0) = input[-1] * mask[0] + input[0] * mask[1] + input[1] * mask[2]
675  //
676  // can be also illustrated by aligning and multiplying
677  // both arrays:
678  // input array (power map); unpadded view
679  // |x=-1|x=0|x=1|x=2|
680  // input array (power map); padded, real view
681  // |x=0 |x=1|x=2|x=3|
682  // mask:
683  // |m=0 |m=1|m=2|
684  //
685  for (mask_i = 0; mask_i < ThermalAnalyzer::THERMAL_MASK_DIM; mask_i++) {
686 
687  // determine power-map index; note that it is not
688  // out of range due to the padded power maps
689  i = x + (mask_i - ThermalAnalyzer::THERMAL_MASK_CENTER);
690 
691  if (ThermalAnalyzer::DBG) {
692 
693  // mass of dbg messages; only for insane
694  // flag
695  if (ThermalAnalyzer::DBG_INSANE) {
696  std::cout << "DBG> y=" << y << ", x=" << x << ", mask_i=" << mask_i << ", i=" << i << std::endl;
697  }
698 
699  if (i < 0 || i >= ThermalAnalyzer::POWER_MAPS_DIM) {
700  std::cout << "DBG> Convolution data error; i out of range (should be limited by x)" << std::endl;
701  }
702  }
703 
704  // convolution; multiplication of mask element and
705  // power-map bin
706  thermal_map_tmp[x][y] +=
707  this->power_maps[layer][i][y].power_density *
708  this->thermal_masks[layer][mask_i];
709  }
710  }
711  }
712  }
713 
714  // continue w/ vertical convolution; here we convolute the temp thermal map (sized
715  // like the padded power map) w/ the thermal masks in order to obtain the final
716  // thermal map (sized like a non-padded power map)
717  for (layer = 0; layer < layers; layer++) {
718 
719  // walk power-map grid for vertical convolution; convolute mask w/ data
720  // obtained by horizontal convolution (thermal_map_tmp)
721  for (x = ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x < ThermalAnalyzer::THERMAL_MAP_DIM + ThermalAnalyzer::POWER_MAPS_PADDED_BINS; x++) {
722 
723  // index for final thermal map, considers padding offset
724  map_x = x - ThermalAnalyzer::POWER_MAPS_PADDED_BINS;
725 
726  for (y = ThermalAnalyzer::POWER_MAPS_PADDED_BINS; y < ThermalAnalyzer::THERMAL_MAP_DIM + ThermalAnalyzer::POWER_MAPS_PADDED_BINS; y++) {
727 
728  // index for final thermal map, considers padding offset
729  map_y = y - ThermalAnalyzer::POWER_MAPS_PADDED_BINS;
730 
731  // perform 1D vertical convolution
732  for (mask_i = 0; mask_i < ThermalAnalyzer::THERMAL_MASK_DIM; mask_i++) {
733 
734  // determine power-map index; note that it is not
735  // out of range due to the temp thermal map (sized
736  // like the padded power map)
737  i = y + (mask_i - ThermalAnalyzer::THERMAL_MASK_CENTER);
738 
739  if (ThermalAnalyzer::DBG) {
740 
741  // mass of dbg messages; only for insane
742  // flag
743  if (ThermalAnalyzer::DBG_INSANE) {
744  std::cout << "DBG> x=" << x << ", y=" << y << ", map_x=" << map_x << ", map_y=" << map_y;
745  std::cout << ", mask_i=" << mask_i << ", i=" << i << std::endl;
746  }
747 
748  if (i < 0 || i >= ThermalAnalyzer::POWER_MAPS_DIM) {
749  std::cout << "DBG> Convolution data error; i out of range (should be limited by y)" << std::endl;
750  }
751  }
752 
753  // convolution; multiplication of mask element and
754  // power-map bin
755  this->thermal_map[map_x][map_y].temp +=
756  thermal_map_tmp[x][i] *
757  this->thermal_masks[layer][mask_i];
758  }
759  }
760  }
761  }
762 
763  // determine max and avg value
764  max_temp = avg_temp = 0.0;
765  for (x = 0; x < ThermalAnalyzer::THERMAL_MAP_DIM; x++) {
766  for (y = 0; y < ThermalAnalyzer::THERMAL_MAP_DIM; y++) {
767  max_temp = std::max(max_temp, this->thermal_map[x][y].temp);
768  avg_temp += this->thermal_map[x][y].temp;
769  }
770  }
771  avg_temp /= std::pow(ThermalAnalyzer::THERMAL_MAP_DIM, 2);
772 
773  // determine cost: max temp estimation, weighted w/ avg temp
774  ret.cost_temp = avg_temp * max_temp;
775  // store max temp
776  ret.max_temp = max_temp;
777  // also store temp offset
778  ret.temp_offset = parameters.temp_offset;
779  // also link whole thermal map to result
780  ret.thermal_map = &this->thermal_map;
781 
782  if (ThermalAnalyzer::DBG_CALLS) {
783  std::cout << "<- ThermalAnalyzer::performPowerBlurring" << std::endl;
784  }
785 }
