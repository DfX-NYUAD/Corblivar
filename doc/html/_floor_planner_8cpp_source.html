212  // for cases w/o alignment,

216  !this->opt_flags.alignment

312  valid_layout_found = best_sol_found = true;

314  }

316

318  // in order to consider updated cost function

321  }
322  // consider next loop iteration

325  }
326  }

333  fitting_layouts_ratio = static_cast<double>(layout_fit_counter) / accepted_ops;

335  else {

337  }
338

340  if (accepted_ops > 0) {
341  avg_cost /= accepted_ops;
342  }
343

348  std::cout << "SA> Step done:" << std::endl;

359 
360  // determine cost terms and overall cost for new best solution;
361  // assume fitting ratio of 1.0 for better comparability of
362  // different solutions and their cost
363  cost = this->evaluateLayout(corb.getAlignments(), 1.0, SA_phase_two);
364 
365  // finally, restore previous layout
366  corb.restoreCBLs();
367 
368  if (SA_phase_two) {

373  }

375  // fixed outline
376  std::cout << "SA> Current max AR mismatch (ideal value is 0.0): " << cost.outline_actual_value << std::endl;

378

382

389  std::cout << "SA> Max routing utilization: " << cost.routing_util_actual_value << std::endl;

413  }
414

417  std::cout << "SA> Avg cost: " << avg_cost << std::endl;
418  std::cout << "SA> SA temp: " << cur_temp << std::endl;

436

449 FloorPlanner::TempPhase FloorPlanner::updateTemp(double& cur_temp, int const& iteration, int const& iteration_first_valid_layout) const {
450  float loop_factor;
451  double prev_temp;
452  TempPhase phase;
453  std::vector<double> prev_avg_cost;
454  double std_dev_avg_cost;
455  unsigned i, temp_schedule_size;
456 
457  prev_temp = cur_temp;
458 
459  // consider reheating in case the SA search has converged in some (possibly local) minima
460  //
461  // determine std dev of avg cost if some previous temperature steps exist
462  temp_schedule_size = this->tempSchedule.size();
463  if (temp_schedule_size >= FloorPlanner::SA_REHEAT_COST_SAMPLES) {
464 
465  for (i = 1; i <= FloorPlanner::SA_REHEAT_COST_SAMPLES; i++) {
466  prev_avg_cost.push_back(this->tempSchedule[temp_schedule_size - i].avg_cost);
467  }
468 
469  std_dev_avg_cost = Math::stdDev(prev_avg_cost);
470  }
471  // else ignore std dev by setting it above limit
472  else {
473  std_dev_avg_cost = FloorPlanner::SA_REHEAT_STD_DEV_COST_LIMIT + 1;
474  }
475 
476  // phase 3; brief reheating due to cost convergence
477  if (std_dev_avg_cost <= FloorPlanner::SA_REHEAT_STD_DEV_COST_LIMIT) {
478 
479  cur_temp *= this->schedule.temp_factor_phase3;
480 
481  phase = TempPhase::PHASE_3;
482  }
483  // phase 1; adaptive cooling (slows down from schedule.temp_factor_phase1 to
484  // schedule.temp_factor_phase1_limit)
485  else if (iteration_first_valid_layout == Point::UNDEF) {
486 
487  loop_factor = (this->schedule.temp_factor_phase1_limit - this->schedule.temp_factor_phase1) *
488  static_cast<float>(iteration - 1) / (this->schedule.loop_limit - 1.0);
489 
490  // note that loop_factor is additive in this case; the cooling factor is
491  // increased w/ increasing iterations
492  cur_temp *= this->schedule.temp_factor_phase1 + loop_factor;
493 
494  phase = TempPhase::PHASE_1;
495  }
496  // phase 2; reheating and converging (initially reheats and then increases cooling
497  // rate faster, i.e., heating factor is decreased w/ increasing iterations to
498  // enable convergence)
499  else {
500  // note that loop_factor must only consider the remaining iteration range
501  loop_factor = 1.0 - static_cast<float>(iteration - iteration_first_valid_layout) /
502  static_cast<float>(this->schedule.loop_limit - iteration_first_valid_layout);
503 
504  cur_temp *= this->schedule.temp_factor_phase2 * loop_factor;
505 
506  phase = TempPhase::PHASE_2;
507  }
508 
509  if (this->logMax()) {
510  std::cout << "SA> (new) temp-update factor: " << cur_temp / prev_temp << " (phase " << phase << ")" << std::endl;
511  }
512 
513  return phase;
514 }

516 void FloorPlanner::initSA(CorblivarCore& corb, std::vector<double>& cost_samples, int& innerLoopMax, double& init_temp) {
517  int i;
518  int accepted_ops;
519  bool op_success;
520  double cur_cost, prev_cost, cost_diff;
521 
522  // reset max cost
523  this->max_cost_WL = 0.0;
524  this->max_cost_routing_util = 0.0;
525  this->max_cost_TSVs = 0;
526  this->max_cost_thermal = 0.0;
527  this->max_cost_alignments = 0.0;
528  this->max_cost_timing = 0.0;
529  this->max_cost_voltage_assignment = 0.0;
530 
531  // reset temperature-schedule log
532  this->tempSchedule.clear();
533 
534  // backup initial CBLs
535  corb.backupCBLs();
536 
537  // init SA parameter: inner loop ops
538  innerLoopMax = std::pow(static_cast<double>(this->blocks.size()), this->schedule.loop_factor);
539 
541  //
542  if (this->logMed()) {
543  std::cout << "SA> Perform initial solution-space sampling..." << std::endl;
544  }
545 
546  // init cost; ignore alignment here
547  this->generateLayout(corb);
548  cur_cost = this->evaluateLayout(corb.getAlignments()).total_cost;
549 
550  // perform some random operations, for SA temperature = 0.0
551  // i.e., consider only solutions w/ improved cost
552  // track acceptance ratio and cost (phase one, area and AR mismatch)
553  // also trigger cost function to assume no fitting layouts
554  i = 1;
555  accepted_ops = 0;
556  cost_samples.reserve(SA_SAMPLING_LOOP_FACTOR * this->blocks.size());
557 
558  while (i <= SA_SAMPLING_LOOP_FACTOR * static_cast<int>(this->blocks.size())) {
559 
560  // trigger random op; assume some fitting layout was found previously such
561  // that not only blocks exceeding the outline are adapted but rather
562  // random operations are performed
563  op_success = layoutOp.performLayoutOp(corb, 1);
564 
565  if (op_success) {
566 
567  prev_cost = cur_cost;
568 
569  // generate layout
570  this->generateLayout(corb);
571  // evaluate layout, new cost
572  cur_cost = this->evaluateLayout(corb.getAlignments()).total_cost;
573  // cost difference
574  cost_diff = cur_cost - prev_cost;
575 
576  // solution w/ worse cost, revert
577  if (cost_diff > 0.0) {
578  // revert last op
579  layoutOp.performLayoutOp(corb, 1, false, true);
580  // reset cost according to reverted CBL
581  cur_cost = prev_cost;
582  }
583  // accept solution w/ improved cost
584  else {
585  // update ops count
586  accepted_ops++;
587  }
588  // store cost
589  cost_samples.push_back(cur_cost);
590 
591  i++;
592  }
593  }
594 
595  // init SA parameter: start temp, depends on std dev of costs [Huan86, see
596  // Shahookar91]
597  init_temp = Math::stdDev(cost_samples) * this->schedule.temp_init_factor;
598 
599  if (this->logMed()) {
600  std::cout << "SA> Done; std dev of cost: " << Math::stdDev(cost_samples) << ", initial temperature: " << init_temp << std::endl;
601  std::cout << "SA> " << std::endl;
602  std::cout << "SA> Perform simulated annealing process..." << std::endl;
603  std::cout << "SA> Phase I: packing blocks into outline..." << std::endl;
604  std::cout << "SA> " << std::endl;
605  }
606 
607  // restore initial CBLs
608  corb.restoreCBLs();
609 }

1065  + this->weights.timing * cost.timing

1092 void FloorPlanner::evaluateTiming(Cost& cost, bool const& set_max_cost, bool reevaluation) {
1093  double max_delay;
1094 
1095  // for reevaluation, after voltage assignment, don't reset voltage-assignment
1096  // values and avoid redundant recalculation of net delays
1097  //
1098  if (!reevaluation) {
1099 
1100  // reset previous voltage assignments if required; they impact the module delay
1101  if (this->opt_flags.voltage_assignment) {
1102 
1103  for (Block& block : this->blocks) {
1104  block.resetVoltageAssignment();
1105  }
1106  }
1107 
1108  // reset previous net delays
1109  for (Net& cur_net : this->nets) {
1110  cur_net.resetSourceMaxDelay();
1111  }
1112 
1113  // assign max net delays to all nets' driving/source blocks; note that some source
1114  // blocks may drive several nets, i.e., the overall max value is finally assigned
1115  for (Net& cur_net : this->nets) {
1116  cur_net.assignSourceMaxDelay();
1117  }
1118  }
1119 
1120  // evaluate max delay over all blocks, drivers and non-driving blocks; covers both
1121  // net delay and actual block delays
1122  //
1123  max_delay = 0.0;
1124  for (Block const& block : this->blocks) {
1125 
1126  max_delay = std::max(max_delay, block.delay());
1127  }
1128 
1129  // memorize max cost; consider given delay threshold
1130  if (set_max_cost) {
1131  this->max_cost_timing = this->IC.delay_threshold;
1132  }
1133 
1134  // apply cost normalization
1135  cost.timing = max_delay / this->max_cost_timing;
1136
