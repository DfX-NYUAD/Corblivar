444  void resetOutline(double TSV_pitch, double width) {
445  double TSV_rows, TSV_cols;
446  double bb_AR;
447  Rect new_bb;
448 
449  // if width is given, orient the island's dimension based on that
450  if (width > 0.0) {
451  new_bb.w = width;
452  new_bb.h = this->TSVs_count * pow(TSV_pitch, 2.0) / width;
453  }
454  // else account for AR of assigned (reference) bb and the number
455  // of TSVs to cover
456  else {
457  // determine number of TSV rows and cols from number of
458  // required TSVs; resembles a square TSV island
459  TSV_rows = std::sqrt(this->TSVs_count);
460  TSV_cols = std::sqrt(this->TSVs_count);
461 
462  // account for AR of given bb
463  bb_AR = this->bb.w / this->bb.h;
464  // however, consider only ``reasonable'' AR limits
465  bb_AR = std::max(bb_AR, TSV_Island::AR_MIN);
466  bb_AR = std::min(bb_AR, TSV_Island::AR_MAX);
467  TSV_rows *= bb_AR;
468  TSV_cols /= bb_AR;
469 
470  // round up rows and cols, spare TSVs are not as ``bad''
471  // as missing TSVs for signal routing; this way it's also
472  // guaranteed that at least one row and col are considered
473  TSV_rows = std::ceil(TSV_rows);
474  TSV_cols = std::ceil(TSV_cols);
475 
476  new_bb.w = TSV_rows * TSV_pitch;
477  new_bb.h = TSV_cols * TSV_pitch;
478  }
479 
480  // calculate island's area
481  new_bb.area = new_bb.w * new_bb.h;
482 
483  // place new bb such into the given bb that their center points
484  // are (roughly) aligned
485  new_bb.ll.x = std::max(0.0, this->bb.ll.x + (this->bb.w - new_bb.w) / 2.0);
486  new_bb.ll.y = std::max(0.0, this->bb.ll.y + (this->bb.h - new_bb.h) / 2.0);
487 
488  // determine new bb's upper bound
489  new_bb.ur.x = new_bb.ll.x + new_bb.w;
490  new_bb.ur.y = new_bb.ll.y + new_bb.h;
491 
492  // replace bb w/ new bb
493  this->bb = std::move(new_bb);
494 
495  // dbg logging for TSV scaling
496  if (TSV_Island::DBG) {
497 
498  std::cout << "DBG_TSVS> TSV group" << std::endl;
499  std::cout << "DBG_TSVS> " << this->id << std::endl;
500  std::cout << "DBG_TSVS> (" << this->bb.ll.x << "," << this->bb.ll.y << ")";
501  std::cout << "(" << this->bb.ur.x << "," << this->bb.ur.y << ")" << std::endl;
502  }
