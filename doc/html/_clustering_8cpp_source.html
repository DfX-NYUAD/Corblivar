307 void Clustering::determineHotspots(ThermalAnalyzer::ThermalAnalysisResult &thermal_analysis) {
308  int x, y;
309  std::vector<ThermalAnalyzer::ThermalMapBin*> thermal_map;
310  std::vector<ThermalAnalyzer::ThermalMapBin*>::iterator it1;
311  std::list<ThermalAnalyzer::ThermalMapBin*> relev_neighbors;
312  std::list<ThermalAnalyzer::ThermalMapBin*>::iterator it2;
313  std::vector<int> neighbors;
314  std::vector<int>::iterator it3;
315  ThermalAnalyzer::ThermalMapBin *cur_bin;
316  int hotspot_id;
317  std::map<double, Hotspot, std::greater<double>> hotspots;
318  std::map<double, Hotspot, std::greater<double>>::iterator it4;
319  Hotspot *cur_hotspot;
320  bool bin_handled;
321 
322  // sanity check for available thermal-analysis result; note that these results are
323  // for example _not_ available during the very first run of SA Phase II where
324  // interconnects (and thus this function) are evaluated before the thermal profile
325  if (thermal_analysis.thermal_map == nullptr) {
326  return;
327  }
328 
329  // reset hotspot regions
330  this->hotspots.clear();
331 
332  // reset hotspot associations in the thermal map
333  for (x = 0; x < ThermalAnalyzer::THERMAL_MAP_DIM; x++) {
334  for (y = 0; y < ThermalAnalyzer::THERMAL_MAP_DIM; y++) {
335  (*thermal_analysis.thermal_map)[x][y].hotspot_id = ThermalAnalyzer::HOTSPOT_UNDEFINED;
336  }
337  }
338 
339  // parse the thermal grid into an list (to be sorted below); data structure for
340  // blob detection
341  for (x = 0; x < ThermalAnalyzer::THERMAL_MAP_DIM; x++) {
342  for (y = 0; y < ThermalAnalyzer::THERMAL_MAP_DIM; y++) {
343 
344  // ignore bins w/ temperature values near the offset
345  if (Math::doubleComp(thermal_analysis.temp_offset, (*thermal_analysis.thermal_map)[x][y].temp)) {
346  continue;
347  }
348 
349  thermal_map.push_back(&(*thermal_analysis.thermal_map)[x][y]);
350  }
351  }
352 
353  // sort list by temperature values
354  sort(thermal_map.begin(), thermal_map.end(),
355  // lambda expression
356  [&](ThermalAnalyzer::ThermalMapBin* b1, ThermalAnalyzer::ThermalMapBin* b2) {
357  return (b1->temp > b2->temp);
358  }
359  );
360 
361  if (Clustering::DBG_HOTSPOT) {
362  std::cout << "DBG_HOTSPOT> bin w/ global max temperature [x][y]: " << thermal_map.front()->x << ", " << thermal_map.front()->y << std::endl;
363  std::cout << "DBG_HOTSPOT> temp: " << thermal_map.front()->temp << std::endl;
364  for (it2 = thermal_map.front()->neighbors.begin(); it2 != thermal_map.front()->neighbors.end(); ++it2) {
365  std::cout << "DBG_HOTSPOT> neighbor bin [x][y]: " << (*it2)->x << ", " << (*it2)->y << std::endl;
366  }
367  }
368 
369  // group the thermal-map list into hotspot regions; perform actual blob detection
370  hotspot_id = 0;
371  for (it1 = thermal_map.begin(); it1 != thermal_map.end(); ++it1) {
372 
373  cur_bin = (*it1);
374 
375  // determine all neighboring bins w/ higher temperature
376  relev_neighbors.clear();
377  for (it2 = cur_bin->neighbors.begin(); it2 != cur_bin->neighbors.end(); ++it2) {
378 
379  if ((*it2)->temp > cur_bin->temp) {
380  relev_neighbors.push_back(*it2);
381  }
382  }
383 
384  // if no such neighbor exits, then the current bin is a local maximum and
385  // will be the seed for a new hotspot/blob
386  if (relev_neighbors.empty()) {
387 
388  // initialize new hotspot
389  this->hotspots.insert( std::pair<int, Hotspot>(
390  // id is the (temporary) key for the map, used for
391  // easier data access during blob detection
392  hotspot_id,
393  // actual hotspot initialization
394  {
395  // peak temp
396  cur_bin->temp,
397  // base-level temp; currently undefined
398  -1.0,
399  // temperature gradient; currently
400  // undefined
401  -1.0,
402  // allocate vector of associated bins;
403  // initialize with cur_bin as first bin of
404  // new hotspot
405  std::vector<ThermalAnalyzer::ThermalMapBin*>(1, cur_bin),
406  // memorize hotspot as still growing
407  true,
408  // id
409  hotspot_id,
410  // score; currently undefined
411  -1.0,
412  // enclosing bb; initialize with cur_bin
413  cur_bin->bb
414  })
415  );
416 
417  // mark bin as associated to this new hotspot
418  cur_bin->hotspot_id = hotspot_id;
419 
420  // increment hotspot counter/id
421  hotspot_id++;
422  }
423 
424  // some neighbor bins w/ higher temperatures exit
425  else {
426  bin_handled = false;
427 
428  // if any of these neighbors is a background bin, then this bin is
429  // also a background bin
430  for (it2 = relev_neighbors.begin(); it2 != relev_neighbors.end(); ++it2) {
431 
432  if ((*it2)->hotspot_id == ThermalAnalyzer::HOTSPOT_BACKGROUND) {
433 
434  cur_bin->hotspot_id = ThermalAnalyzer::HOTSPOT_BACKGROUND;
435  bin_handled = true;
436 
437  break;
438  }
439  }
440 
441  // none of the neighbors are background bins; proceed with check
442  // if the neighbors belong to one or to different hotspots
443  if (!bin_handled) {
444 
445  neighbors.clear();
446 
447  for (it2 = relev_neighbors.begin(); it2 != relev_neighbors.end(); ++it2) {
448 
449  if (Clustering::DBG_HOTSPOT) {
450 
451  if ((*it2)->hotspot_id == ThermalAnalyzer::HOTSPOT_UNDEFINED) {
452 
453  std::cout << "DBG_HOTSPOT> blob-detection error; undefined bin triggered" << std::endl;
454  continue;
455  }
456  }
457 
458  neighbors.push_back((*it2)->hotspot_id);
459  }
460 
461  if (Clustering::DBG_HOTSPOT) {
462 
463  if (neighbors.empty()) {
464  std::cout << "DBG_HOTSPOT> blob-detection error; no valid neighbor bin found" << std::endl;
465  }
466  }
467 
468  // memorize only unique bins
469  sort(neighbors.begin(), neighbors.end());
470  unique(neighbors.begin(), neighbors.end());
471 
472  // all neighbors belong to one specific hotspot
473  if (neighbors.size() == 1) {
474 
475  cur_hotspot = &this->hotspots.find(neighbors.front())->second;
476 
477  // if the hotspot is allowed to grow, associated
478  // this bin with it, and mark bin as well
479  if (cur_hotspot->still_growing) {
480 
481  cur_hotspot->bins.push_back(cur_bin);
482  cur_bin->hotspot_id = cur_hotspot->id;
483  }
484  // if the hotspot is not allowed to grow anymore,
485  // mark the bin as background bin
486  else {
487  cur_bin->hotspot_id = ThermalAnalyzer::HOTSPOT_BACKGROUND;
488  }
489  }
490  // neighbors belong to different hotspots
491  else {
492  // the bin has to be background since it defines
493  // the base level for different hotspots
494  cur_bin->hotspot_id = ThermalAnalyzer::HOTSPOT_BACKGROUND;
495 
496  // the different hotspots have reached their base
497  // level w/ this bin; mark them as not growing
498  // anymore and memorize the base-level temp
499  for (it3 = neighbors.begin(); it3 != neighbors.end(); ++it3) {
500 
501  this->hotspots.find(*it3)->second.still_growing = false;
502  this->hotspots.find(*it3)->second.base_temp = cur_bin->temp;
503 
504  // the determination of temp gradient and
505  // score could be also conducted here, but
506  // is postponed since a post-processing of
507  // all hotspot regions is required anyway
508  }
509  }
510  }
511  }
512  }
513 
514  // post-processing hotspot regions, also re-order them according to their score
515  for (it4 = this->hotspots.begin(); it4 != this->hotspots.end(); ++it4) {
516 
517  cur_hotspot = &(*it4).second;
518 
519  // some regions may be still marked as growing; mark such regions as not
520  // growing anymore
521  if (cur_hotspot->still_growing) {
522 
523  cur_hotspot->still_growing = false;
524 
525  // also approximate base temp, using the minimal temperature of
526  // all bins of the hotspot; note that the actual base temp is
527  // slightly lower since the base-level bin is not included in the
528  // hotspot itself
529  cur_hotspot->base_temp = (*cur_hotspot->bins.begin())->temp;
530  for (it1 = cur_hotspot->bins.begin(); it1 != cur_hotspot->bins.end(); ++it1) {
531 
532  cur_hotspot->base_temp = std::min(cur_hotspot->base_temp, (*it1)->temp);
533  }
534  }
535 
536  // using the base temp, determine gradient
537  cur_hotspot->temp_gradient = cur_hotspot->peak_temp - cur_hotspot->base_temp;
538 
539  // determine hotspot score; the score is defined by its peak temp and temp
540  // gradient, i.e., measures how ``critical'' the local thermal maxima is
541  cur_hotspot->score = cur_hotspot->temp_gradient * std::pow(cur_hotspot->peak_temp, 2.0);
542  // apply normalization such that scores are roughly in the range of
543  // [0..10]
544  cur_hotspot->score /= Clustering::SCORE_NORMALIZATION;
545 
546  // determine the (all bins enclosing) bb; this is used to simplify checks
547  // of nets overlapping hotspot regions, but also reduces spatial accuracy
548  for (it1 = cur_hotspot->bins.begin(); it1 != cur_hotspot->bins.end(); ++it1) {
549 
550  cur_hotspot->bb = Rect::determBoundingBox(cur_hotspot->bb, (*it1)->bb);
551  }
552 
553  // enlarge final bb by 2x, which should increase chances for the
554  // subsequent clustering to match net bounding box with these cluster bbs
555  cur_hotspot->bb.ll.x -= cur_hotspot->bb.w / 2.0;
556  cur_hotspot->bb.ur.x += cur_hotspot->bb.w / 2.0;
557  cur_hotspot->bb.ll.y -= cur_hotspot->bb.h / 2.0;
558  cur_hotspot->bb.ur.y += cur_hotspot->bb.h / 2.0;
559  cur_hotspot->bb.w = cur_hotspot->bb.ur.x - cur_hotspot->bb.ll.x;
560  cur_hotspot->bb.h = cur_hotspot->bb.ur.y - cur_hotspot->bb.ll.y;
561  cur_hotspot->bb.area = cur_hotspot->bb.w * cur_hotspot->bb.h;
562 
563  // put hotspot into (temporary) map, which is sorted by the hotspot scores
564  // and later replaces the global map
565  hotspots.insert( std::pair<double, Hotspot>(
566  cur_hotspot->score,
567  std::move(*cur_hotspot)
568  ));
569  }
570 
571  // replace global map w/ new sorted map
572  this->hotspots = std::move(hotspots);
573 
574  if (Clustering::DBG_HOTSPOT) {
575  int bins_hotspot = 0;
576  int bins_background = 0;
577  int bins_undefined = 0;
578 
579  std::cout << "DBG_HOTSPOT> hotspots :" << std::endl;
580 
581  for (it4 = this->hotspots.begin(); it4 != this->hotspots.end(); ++it4) {
582  std::cout << "DBG_HOTSPOT> id: " << (*it4).second.id << std::endl;
583  std::cout << "DBG_HOTSPOT> bb: (" << (*it4).second.bb.ll.x << "," << (*it4).second.bb.ll.y;
584  std::cout << "),(" << (*it4).second.bb.ur.x << "," << (*it4).second.bb.ur.y << ")" << std::endl;
585  std::cout << "DBG_HOTSPOT> peak temp: " << (*it4).second.peak_temp << std::endl;
586  std::cout << "DBG_HOTSPOT> base temp: " << (*it4).second.base_temp << std::endl;
587  std::cout << "DBG_HOTSPOT> temp gradient: " << (*it4).second.temp_gradient << std::endl;
588  std::cout << "DBG_HOTSPOT> score: " << (*it4).second.score << std::endl;
589  std::cout << "DBG_HOTSPOT> bins count: " << (*it4).second.bins.size() << std::endl;
590  std::cout << "DBG_HOTSPOT> still growing: " << (*it4).second.still_growing << std::endl;
591  }
592 
593  std::cout << "DBG_HOTSPOT> adapted thermal-map:" << std::endl;
594 
595  for (x = 0; x < ThermalAnalyzer::THERMAL_MAP_DIM; x++) {
596  for (y = 0; y < ThermalAnalyzer::THERMAL_MAP_DIM; y++) {
597
