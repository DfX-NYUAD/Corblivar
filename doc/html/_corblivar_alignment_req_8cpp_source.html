38  // initially, assume the request to be feasible

42  this->s_i->alignment = Block::AlignmentStatus::SUCCESS;
43  this->s_j->alignment = Block::AlignmentStatus::SUCCESS;
44 
45  // for request w/ alignment ranges, we verify the alignment via the
46  // blocks' intersection
47  if (this->range_x() || this->range_y()) {
48  blocks_intersect = Rect::determineIntersection(this->s_i->bb, this->s_j->bb);
49  }
50  // for requests w/ max distance ranges, we verify the alignment via the
51  // blocks' bounding box (considering the blocks' center points)
52  if (this->range_max_x() || this->range_max_y()) {
53  blocks_bb = Rect::determBoundingBox(this->s_i->bb, this->s_j->bb, true);
54  }
55 
56  // check partial request, horizontal alignment
57  //
58  // alignment range
59  if (this->range_x()) {
60 
61  // consider the spatial mismatch as cost; overlap too small
62  if (blocks_intersect.w < this->alignment_x) {
63 
64  // missing overlap
65  ret.cost += this->alignment_x - blocks_intersect.w;
66 
67  // in case blocks don't overlap at all, also consider the
68  // blocks' distance as further cost
69  if (blocks_intersect.w == 0) {
70 
71  if (Rect::rectA_leftOf_rectB(this->s_i->bb, this->s_j->bb, false)) {
72 
73  ret.cost += this->s_j->bb.ll.x - this->s_i->bb.ur.x;
74 
75  // annotate block-alignment failure
76  this->s_i->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_LEFT;
77  this->s_j->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_RIGHT;
78  }
79  else {
80 
81  ret.cost += this->s_i->bb.ll.x - this->s_j->bb.ur.x;
82 
83  // annotate block-alignment failure
84  this->s_i->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_RIGHT;
85  this->s_j->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_LEFT;
86  }
87  }
88 
89  // annotate general alignment failure
90  this->fulfilled = false;
91  }
92  }
93  // max distance range
94  else if (this->range_max_x()) {
95 
96  // consider the spatial mismatch as cost; distance too large
97  if (blocks_bb.w > this->alignment_x) {
98 
99  ret.cost += blocks_bb.w - this->alignment_x;
100 
101  // annotate general alignment failure
102  this->fulfilled = false;
103 
104  // annotate block-alignment failure
105  if (this->s_i->bb.ll.x < this->s_j->bb.ll.x) {
106  this->s_i->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_LEFT;
107  this->s_j->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_RIGHT;
108  }
109  else {
110  this->s_i->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_RIGHT;
111  this->s_j->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_LEFT;
112  }
113  }
114  }
115  // fixed alignment offset
116  else if (this->offset_x()) {
117 
118  // check the blocks' offset against the required offset
119  if (!Math::doubleComp(this->s_j->bb.ll.x - this->s_i->bb.ll.x, this->alignment_x)) {
120 
121  // s_j should be to the right of s_i;
122  // consider the spatial mismatch as cost
123  if (this->alignment_x >= 0.0) {
124 
125  // s_j is to the right of s_i
126  if (this->s_j->bb.ll.x > this->s_i->bb.ll.x) {
127 
128  // abs required for cases where s_j is too
129  // far left, i.e., not sufficiently away

137  this->s_j->alignment = Block::AlignmentStatus::FAIL_HOR_TOO_LEFT;
138  }

368 bool CorblivarAlignmentReq::vertical_bus() const {
369  return (
370  // min overlap in both dimensions
371  (this->range_x() && this->range_y()) ||
372  // zero-offset fixed alignment in both dimensions
373  (this->offset_x() && this->alignment_x == 0 && this->offset_y() && this->alignment_y == 0) ||
374  // non-zero offset in both dimensions;
375  (this->offset_x() && this->alignment_x != 0.0 && this->offset_y() && this->alignment_y != 0.0
376  // but with sufficiently small offset such that blocks will
377  // partially intersect
378  && (
379  // positive offset: b_j not further offset to the right/top than
380  // b_i is wide/high; negative offset: b_j not further offset to
381  // the left/bottom than b_j is wide/high
382  (this->alignment_x > 0.0) ? this->alignment_x < this->s_i->bb.w : this->alignment_x > -(this->s_j->bb.w) &&
383  (this->alignment_y > 0.0) ? this->alignment_y < this->s_i->bb.h : this->alignment_y > -(this->s_j->bb.h)
384  )
385  )
386  );
387 }
