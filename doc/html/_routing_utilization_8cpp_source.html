87 RoutingUtilization::UtilResult RoutingUtilization::determCost() const {
88  unsigned x, y;
89  unsigned layer;
90  UtilResult ret;
91 
92  ret.cost = ret.avg_util = ret.max_util = 0.0;
93 
94  for (layer = 0; layer < this->util_maps.size(); layer++) {
95  for (x = 0; x < this->util_maps[0].size(); x++) {
96  for (y = 0; y < this->util_maps[0][0].size(); y++) {
97 
98  // determine max util
99  if (this->util_maps[layer][x][y].utilization > ret.max_util) {
100  ret.max_util = this->util_maps[layer][x][y].utilization;
101  }
102 
103  // sum up util, required for avg util
104  ret.avg_util += this->util_maps[layer][x][y].utilization;
105  }
106  }
107  }
108 
109  ret.avg_util /= this->util_maps.size();
110  ret.avg_util /= this->util_maps[0].size();
111  ret.avg_util /= this->util_maps[0][0].size();
112 
113  // cost: avg and max util
114  ret.cost = ret.avg_util * ret.max_util;
115 
116  return ret;
117 }

119 void RoutingUtilization::adaptUtilMap(int const& layer, Rect const& net_bb, double const& net_weight) {
120  int x, y;
121  double util;
122  int x_lower, x_upper, y_lower, y_upper;
123  Rect bb_ext;
124 
125  if (RoutingUtilization::DBG_CALLS) {
126  std::cout << "-> RoutingUtilization::adaptUtilMap(" << layer << ", " << net_bb.area << ", " << net_weight << ")" << std::endl;
127  }
128 
129  // determine index boundaries for utilization map; based on intersection of map and
130  // net's bb; note that cast to int truncates toward zero, i.e., performs like
131  // floor for positive numbers
132  x_lower = static_cast<int>(net_bb.ll.x / this->util_maps_dim_x);
133  y_lower = static_cast<int>(net_bb.ll.y / this->util_maps_dim_y);
134  // +1 in order to efficiently emulate the result of ceil(); limit upper
135  // bound to util-maps dimensions
136  x_upper = std::min(static_cast<int>(net_bb.ur.x / this->util_maps_dim_x) + 1, RoutingUtilization::UTIL_MAPS_DIM);
137  y_upper = std::min(static_cast<int>(net_bb.ur.y / this->util_maps_dim_y) + 1, RoutingUtilization::UTIL_MAPS_DIM);
138 
139  if (RoutingUtilization::DBG) {
140  std::cout << "DBG_ROUTING_UTIL> Affected util-map bins: " << x_lower << "," << y_lower
141  << " to " <<
142  x_upper << "," << y_upper << std::endl;
143  }
144 
145  // simple routing-utilization model: even distribution, as discussed in
146  // [Meister11]; this model is surprisingly accurate for practical benchmarks;
147  // calculate utilization according to wirelength, covered area, and net weight
148  //
149  // for area and wirelength, we consider the (by above floor and ceil index
150  // boundaries) slightly extended bb
151  bb_ext.w = this->util_maps_bins_ll_x[x_upper] - this->util_maps_bins_ll_x[x_lower];
152  bb_ext.h = this->util_maps_bins_ll_y[y_upper] - this->util_maps_bins_ll_y[y_lower];
153  bb_ext.area = bb_ext.w * bb_ext.h;
154 
155  util = net_weight * ((bb_ext.w + bb_ext.h) / bb_ext.area);
156 
157  // walk util-map bins covering intersection; adapt routing utilization
158  for (x = x_lower; x < x_upper; x++) {
159  for (y = y_lower; y < y_upper; y++) {
160 
161  // adapt map on affected layer
162  this->util_maps[layer][x][y].utilization += util;
163  }
164  }
165 
166  if (RoutingUtilization::DBG_CALLS) {
167  std::cout << "<- RoutingUtilization::adaptPowerMaps" << std::endl;
168  }
169 }
