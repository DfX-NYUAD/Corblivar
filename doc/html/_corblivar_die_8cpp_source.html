33 void CorblivarDie::placeCurrentBlock(bool const& alignment_enabled) {
34  std::list<Block const*> relevBlocks;
35 
36  // current tuple; only mutable block parameters can be edited
37  Block const* cur_block = this->getCurrentBlock();
38 
39  // sanity check for previously placed blocks; may occur due to multiple alignment
40  // requests in process covering this particular block
41  if (cur_block->placed) {
42  return;
43  }
44 
45  // pop relevant blocks from related placement stack
46  relevBlocks = this->popRelevantBlocks();
47 
48  // horizontal placement
49  if (this->getCurrentDirection() == Direction::HORIZONTAL) {
50 
51  // first, determine block's y-coordinates
52  this->determCurrentBlockCoords(Coordinate::Y, relevBlocks);
53  // second, determine block's x-coordinates (depends on y-coord; extended
54  // check depends on whether alignment is enabled, i.e., whether some
55  // blocks might be shifted)
56  this->determCurrentBlockCoords(Coordinate::X, relevBlocks, alignment_enabled);
57  }
58  // vertical placement
59  else {
60 
61  // first, determine block's x-coordinates
62  this->determCurrentBlockCoords(Coordinate::X, relevBlocks);
63  // second, determine block's y-coordinates (depends on x-coord; extended
64  // check depends on whether alignment is enabled, i.e., whether some
65  // blocks might be shifted)
66  this->determCurrentBlockCoords(Coordinate::Y, relevBlocks, alignment_enabled);
67  }
68 
69  // update placement stacks
70  this->updatePlacementStacks(relevBlocks);
71 
72  // mark block as placed
73  cur_block->placed = true;
74 
75  // placement stacks debugging
76  if (CorblivarDie::DBG_STACKS) {
77  this->debugStacks();
78  }
79 }

145 std::list<Block const*> CorblivarDie::popRelevantBlocks() {
146  std::list<Block const*> ret;
147  unsigned blocks_count;
148 
149  // horizontal placement; consider stack Hi
150  if (this->getCurrentDirection() == Direction::HORIZONTAL) {
151 
152  // relevant blocks count depends on the T-junctions to be covered and the
153  // current stack itself
154  blocks_count = std::min<unsigned>(this->getJunctions(this->pi) + 1, this->Hi.size());
155 
156  // pop relevant blocks from stack into return list
157  while (blocks_count > ret.size()) {
158  ret.push_back(std::move(this->Hi.front()));
159  this->Hi.pop_front();
160  }
161  }
162  // vertical placement; consider stack Vi
163  else {
164 
165  // relevant blocks count depends on the T-junctions to be covered and the
166  // current stack itself
167  blocks_count = std::min<unsigned>(this->getJunctions(this->pi) + 1, this->Vi.size());
168 
169  // pop relevant blocks from stack into return list
170  while (blocks_count > ret.size()) {
171  ret.push_back(std::move(this->Vi.front()));
172  this->Vi.pop_front();
173  }
174  }
175 
176  return ret;
177 }

180  bool add_to_stack;
181  Block const* b;
182 
183  // current block
184  Block const* cur_block = this->getCurrentBlock();
185  // current block's insertion direction
186  Direction const& cur_dir = this->getCurrentDirection();
187 
188  if (CorblivarDie::DBG_STACKS) {
189  std::cout << "DBG_CORB> Update stacks; current block: " << cur_block->id << "; block's dir: " << static_cast<unsigned>(cur_dir);
190 
191  std::cout << "; relevant blocks: ";
192  for (Block const* b : relev_blocks_stack) {
193  if (b->numerical_id != relev_blocks_stack.back()->numerical_id) {
194  std::cout << b->id << ", ";
195  }
196  else {
197  std::cout << b->id;
198  }
199  }
200 
201  std::cout << std::endl;
202  }
203 
204  // horizontal placement
205  if (cur_dir == Direction::HORIZONTAL) {
206 
207  // update vertical stack; add cur_block when no other relevant blocks
208  // are to its top side
209  //
210  // note that this check is indepent of overlap in x-direction; this way,
211  // we avoid situations where different blocks on both stacks Hi/Vi are
212  // current corner block which would result in invalid layouts

214  for (Block const* b : relev_blocks_stack) {

218  }

222  this->Vi.push_front(cur_block);
223  }
224

226  //

229  //
230  // add relevant blocks which have no block to the right, simplified by

232  // by reverse iteration, we retain the (implicit) ordering of blocks
233  // popped from stack Hi regarding their insertion order; required for
234  // proper stack manipulation
235  for (std::list<Block const*>::reverse_iterator r_iter = relev_blocks_stack.rbegin(); r_iter != relev_blocks_stack.rend(); ++r_iter) {
236  b = *r_iter;
237 
238  if (!Rect::rectA_leftOf_rectB(b->bb, cur_block->bb, true)) {
239  this->Hi.push_front(b);
240  }
241  }
242  }
243  // vertical placement
244  else {
245 
246  // update horizontal stack; add cur_block when no other relevant blocks
247  // are to its right side
248  //
249  // note that this check is indepent of overlap in y-direction; this way,
250  // we avoid situations where different blocks on both stacks Hi/Vi are
251  // current corner block which would result in invalid layouts
252  add_to_stack = true;
253  for (Block const* b : relev_blocks_stack) {
254  if (Rect::rectA_leftOf_rectB(cur_block->bb, b->bb, false)) {
255  add_to_stack = false;
256  break;
257  }
258  }

261  this->Hi.push_front(cur_block);

263

266  // always consider cur_block since it's one of the top-most blocks now
267  this->Vi.push_front(cur_block);

271  // iteration, we retain the (implicit) ordering of blocks popped from
272  // stack Vi regarding their insertion order; required for proper stack

284 void CorblivarDie::rebuildPlacementStacks(std::list<Block const*>& relev_blocks_stack) {
285  std::list<Block const*>::iterator iter;
286  std::list<Block const*> stack_backup;
287  bool covered;
288 
289  // current block
290  Block const* cur_block = this->getCurrentBlock();
291  // current block's insertion direction
292  Direction const& cur_dir = this->getCurrentDirection();
293 
294  if (CorblivarDie::DBG_STACKS) {
295  std::cout << "DBG_CORB> Rebuild stacks; current block: " << cur_block->id << ", block's dir: " << static_cast<unsigned>(cur_dir);
296 
297  std::cout << "; relevant blocks: ";
298  for (Block const* b : relev_blocks_stack) {
299  if (b->numerical_id != relev_blocks_stack.back()->numerical_id) {
300  std::cout << b->id << ", ";
301  }
302  else {
303  std::cout << b->id;
304  }
305  }
306 
307  std::cout << std::endl;
308  }
309 
310  // after block shifting, we cannot easily make assumption on the resulting layout
311  // and thus how to update the stacks (note that even the insertion direction may
312  // be implicitly different after shifting); thus, we simply a) update the stacks
313  // w/ all relevant, uncovered blocks (includes previously popped blocks and blocks
314  // still on stack), b) push back the current block itself and c) sort the stacks
315  // by related dimensions, i.e., block-insertion order; note that this process has
316  // to be done for both stacks
317 
318  // horizontal stack Hi
319  //
320  // a) check remaining blocks if they are covered now (by current block)
321  for (iter = this->Hi.begin(); iter != this->Hi.end();) {
322 
323  covered = false;
324 
325  // block is now covered
326  if (Rect::rectA_leftOf_rectB((*iter)->bb, cur_block->bb, true)) {
327 
328  covered = true;
329 
330  // drop block from stack; also update iter to point to next
331  // element
332  iter = this->Hi.erase(iter);
333  }
334 
335  // regular iter update
336  if (!covered) {
337  ++iter;
338  }
339  }
340 
341  // a) push back relevant blocks in case they are not covered (by current block);
342  // only for related insertion direction
343  if (cur_dir == Direction::HORIZONTAL) {
344 
345  for (Block const* b : relev_blocks_stack) {
346  if (!Rect::rectA_leftOf_rectB(b->bb, cur_block->bb, true)) {
347  this->Hi.push_front(b);
348  }
349  }
350  }
351 
352  // b) push back the current block itself; only if not covered by any placed block
353  covered = false;
354  // walk all blocks (implicitly ordered such that placed blocks are first)
355  for (unsigned b = 0; b < this->getCBL().size(); b++) {
356 
357  // if not yet placed block is reached, the following blocks are also not
358  // placed, i.e., not relevant; break loop
359  if (!this->getBlock(b)->placed) {
360  break;
361  }
362  else {
363  if (Rect::rectA_leftOf_rectB(cur_block->bb, this->getBlock(b)->bb, true)) {
364  covered = true;
365  break;
366  }
367  }
368  }
369  if (!covered) {
370  this->Hi.push_back(cur_block);
371  }
372 
373  // c) sort stack by y-dimension in descending order; retains the proper stack
374  // structure for further horizontal block insertion
375  this->Hi.sort(
376  // lambda expression
377  [&](Block const* b1, Block const* b2) {
378  // descending order, b1 above b2
379  return !Rect::rectA_below_rectB(b1->bb, b2->bb, false);
380  }
381  );
382 
383  // vertical stack Vi
384  //
385  // a) check remaining blocks if they are covered now (by current block)
386  for (iter = this->Vi.begin(); iter != this->Vi.end();) {
387 
388  covered = false;
389 
390  // block is now covered
391  if (Rect::rectA_below_rectB((*iter)->bb, cur_block->bb, true)) {
392 
393  covered = true;
394 
395  // drop block from stack; also update iter to point to next
396  // element
397  iter = this->Vi.erase(iter);
398  }
399 
400  // regular iter update
401  if (!covered) {
402  ++iter;
403  }
404  }
405 
406  // a) push back relevant blocks in case they are not covered (by current block);
407  // only for related insertion direction
408  if (cur_dir == Direction::VERTICAL) {
409 
410  for (Block const* b : relev_blocks_stack) {
411  if (!Rect::rectA_below_rectB(b->bb, cur_block->bb, true)) {
412  this->Vi.push_front(b);
413  }
414  }
415  }
416 
417  // b) push back the current block itself; only if not covered by any placed block
418  covered = false;
419  // walk all blocks (implicitly ordered such that placed blocks are first)
420  for (unsigned b = 0; b < this->getCBL().size(); b++) {
421 
422  // if not yet placed block is reached, the following blocks are also not
423  // placed, i.e., not relevant; break loop
424  if (!this->getBlock(b)->placed) {
425  break;
426  }
427  else {
428  if (Rect::rectA_below_rectB(cur_block->bb, this->getBlock(b)->bb, true)) {
429  covered = true;
430  break;
431  }
432  }
433  }
434  if (!covered) {
435  this->Vi.push_back(cur_block);
436  }
437 
438  // c) sort stack by x-dimension in descending order; retains the proper stack
439  // structure for further vertical block insertion
440  this->Vi.sort(
441  // lambda expression
442  [&](Block const* b1, Block const* b2) {
443  // descending order, b1 right of b2
444  return !Rect::rectA_leftOf_rectB(b1->bb, b2->bb, false);
445  }
446  );
447 
448  // sanity check for different corner blocks; may result due to shifting of blocks;
449  // we need to try fixing both stacks since we cannot assume which is the correct
450  // corner block in this case
451  if (this->Hi.front() != this->Vi.front()) {
452 
453  // first, try to fix Hi
454  //
455  // local copy Hi for backup
456  stack_backup = this->Hi;
457 
458  // try dropping blocks until corner blocks match
459  while (this->Hi.front() != this->Vi.front()) {
460 
461  if (this->Hi.empty()) {
462  break;
463  }
464  else {
465  this->Hi.pop_front();
466  }
467  }
468 
469  // fixing this stack failed, retry w/ Vi
470  if (this->Hi.empty()) {
471 
472  // restore Hi
473  this->Hi = stack_backup;
474 
475  // local copy Vi for backup
476  stack_backup = this->Vi;
477 
478  // try dropping blocks until corner blocks match
479  while (this->Hi.front() != this->Vi.front()) {
480 
481  if (this->Vi.empty()) {
482  break;
483  }
484  else {
485  this->Vi.pop_front();
486  }
487  }
488 
489  // 2nd stack fix failed; this will most likely result in invalid
490  // layouts
491  if (this->Vi.empty()) {
492 
493  // restore Vi
494  this->Vi = stack_backup;
495 
496  // dbg log for failure
497  if (CorblivarDie::DBG_STACKS) {
498  std::cout << "DBG_CORB> Differing corner blocks on Hi, Vi; related stack fixing failed!" << std::endl;
499  }
500  }
501  }
502  }

505 void CorblivarDie::determCurrentBlockCoords(Coordinate const& coord, std::list<Block const*> const& relev_blocks_stack, bool const& extended_check) const {
506  double x, y;
507 
508  // current block
509  Block const* cur_block = this->getCurrentBlock();
510  // current block's insertion direction
511  Direction const& cur_dir = this->getCurrentDirection();
512 
513  // update x-coordinates
514  if (coord == Coordinate::X) {
515 
516  // for vertical block insertion; x-coordinate is first coordinate, thus
517  // not dependent on y-coordinate
518  if (cur_dir == Direction::VERTICAL) {
519 
520  // determine x-coordinate for lower left corner of current block
521  //
522  // all columns are to be covered (according to T-juncts), thus place the
523  // block at the left die boundary
524  if (this->Vi.empty()) {
525  x = 0;
526  }
527  // only some columns are to be covered, thus determine the left front of
528  // the related blocks
529  else {
530  x = -1;
531  for (Block const* b : relev_blocks_stack) {
532  if (x == -1) {
533  x = b->bb.ll.x;
534  }
535  else {
536  x = std::min(x, b->bb.ll.x);
537  }
538  }
539  }
540  }
541  // for horizontal insertion; x-coordinate is second coordinate, thus
542  // dependent on y-coordinate
543  else {
544  x = 0;
545 
546  // extended check in case of block alignment is required: for
547  // shifted blocks, and generally for all blocks to be placed w/
548  // alignment enabled (relying on possibly shifted blocks given on
549  // the placement stacks), we cannot assume that the current stacks
550  // block are the relevant boundaries; thus, we need to check
551  // against all previously placed blocks
552  //
553  if (extended_check) {
554 
555  // walk all blocks (implicitly ordered such that placed
556  // blocks are first)
557  for (unsigned b = 0; b < this->getCBL().size(); b++) {
558 
559  // if not yet placed block is reached, the
560  // following blocks are also not placed, i.e., not
561  // relevant; break loop
562  if (!this->getBlock(b)->placed) {
563  break;
564  }
565  else {
566  // only consider blocks which intersect in y-direction
567  if (Rect::rectsIntersectVertical(cur_block->bb, this->getBlock(b)->bb)) {
568  // determine right front
569  x = std::max(x, this->getBlock(b)->bb.ur.x);
570  }
571  }
572  }
573  }
574  // non shifted block / trivial case w/o alignment; simply check
575  // against blocks to be covered
576  //
577  else {
578  // determine x-coordinate for lower left corner of current block, consider
579  // right front of blocks to be covered
580  for (Block const* b : relev_blocks_stack) {
581  // only consider blocks which intersect in y-direction
582  if (Rect::rectsIntersectVertical(cur_block->bb, b->bb)) {
583  // determine right front
584  x = std::max(x, b->bb.ur.x);
585  }
586  }
587  }
588  }
589 
590  // update block's x-coordinates
591  cur_block->bb.ll.x = x;
592  cur_block->bb.ur.x = cur_block->bb.w + x;
593  }
594 
595  // update y-coordinates
596  else {
597 
598  // for horizontal block insertion; y-coordinate is first coordinate, thus
599  // not dependent on x-coordinate
600  if (cur_dir == Direction::HORIZONTAL) {
601 
602  // determine y-coordinate for lower left corner of current block
603  //
604  // all rows are to be covered (according to T-juncts), thus place the
605  // block at the bottom die boundary
606  if (this->Hi.empty()) {
607  y = 0;
608  }
609  // only some rows are to be covered, thus determine the lower front of
610  // the related blocks
611  else {
612  y = -1;
613  for (Block const* b : relev_blocks_stack) {
614  if (y == -1) {
615  y = b->bb.ll.y;
616  }
617  else {
618  y = std::min(y, b->bb.ll.y);
619  }
620  }
621  }
622  }
623  // for vertical insertion; y-coordinate is second coordinate, thus
624  // dependent on x-coordinate
625  else {
626  y = 0;
627 
628  // extended check in case of block alignment is required: for
629  // shifted blocks, and generally for all blocks to be placed w/
630  // alignment enabled (relying on possibly shifted blocks given on
631  // the placement stacks), we cannot assume that the current stacks
632  // block are the relevant boundaries; thus, we need to check
633  // against all previously placed blocks
634  //
635  if (extended_check) {
636 
637  // walk all blocks (implicitly ordered such that placed
638  // blocks are first)
639  for (unsigned b = 0; b < this->getCBL().size(); b++) {
640 
641  // if not yet placed block is reached, the
642  // following blocks are also not placed, i.e., not
643  // relevant; break loop
644  if (!this->getBlock(b)->placed) {
645  break;
646  }
647  else {
648  // only consider blocks which intersect in x-direction
649  if (Rect::rectsIntersectHorizontal(cur_block->bb, this->getBlock(b)->bb)) {
650  // determine upper front
651  y = std::max(y, this->getBlock(b)->bb.ur.y);
652  }
653  }
654  }
655  }
656  // non shifted block / trivial case w/o alignment; simply check
657  // against blocks to be covered
658  //
659  else {
660  // determine y-coordinate for lower left corner of current block, consider
661  // upper front of blocks to be covered
662  for (Block const* b : relev_blocks_stack) {
663  // only consider blocks which intersect in x-direction
664  if (Rect::rectsIntersectHorizontal(cur_block->bb, b->bb)) {
665  // determine upper front
666  y = std::max(y, b->bb.ur.y);
667  }
668  }
669  }
670  }
671 
672  // update block's y-coordinates
673  cur_block->bb.ll.y = y;
674  cur_block->bb.ur.y = cur_block->bb.h + y;
675  }

678 bool CorblivarDie::shiftCurrentBlock(Direction const& dir, CorblivarAlignmentReq const* req, bool const& dry_run) {
679  Block const* reference_block;
680  double overlap_offset_x, overlap_offset_y;
681  double shift_x, shift_y;
682  double range_x, range_y;
683  double neighbor_x, neighbor_y;
684  bool neighbor_x_found, neighbor_y_found;
685  bool shifted;
686 
687  // block to shift is current block
688  Block const* shift_block = this->getCurrentBlock();
689 
690  // flag for monitoring shifting itself
691  shifted = false;
692 
693  // first, determine reference block
694  if (shift_block->numerical_id == req->s_i->numerical_id) {
695  reference_block = req->s_j;
696  }
697  else {
698  reference_block = req->s_i;
699  }
700 
701  // second, perform actual shift
702  //
703  // shift in horizontal direction
704  if (dir == Direction::HORIZONTAL) {
705

945  // lambda expression
