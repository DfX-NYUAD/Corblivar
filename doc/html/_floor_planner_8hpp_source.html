418  inline Point shrinkDieOutlines() {
419  Point outline;
420 
421  // determine blocks outline across; reasonable common die outline
422  // for whole 3D-IC stack
423  for (Block const& b : this->blocks) {
424  outline.x = std::max(outline.x, b.bb.ur.x);
425  outline.y = std::max(outline.y, b.bb.ur.y);
426  }
427 
428  // now, shrink fixed outline if any dimension is smaller than
429  // previous outline
430  if (
431  (outline.x < this->IC.outline_x || outline.y < this->IC.outline_y) &&
432  // only if current blocks-outline is smaller than previous outline
433  (outline.x * outline.y) < (this->IC.die_area)
434  ) {
435 
436  this->IC.outline_x = outline.x;
437  this->IC.outline_y = outline.y;
438 
439  // also reset the power maps setting
440  this->thermalAnalyzer.initPowerMaps(this->IC.layers, this->getOutline());
441 
442  // and the routing-estimation maps have to be reset as well
443  this->routingUtil.initUtilMaps(this->IC.layers, this->getOutline());
444 
445  // reset related die properties
446  this->IC.die_AR = this->IC.outline_x / this->IC.outline_y;
447  this->IC.die_area = this->IC.outline_x * this->IC.outline_y;
448  this->IC.stack_area = this->IC.layers * this->IC.die_area;
449  this->IC.stack_deadspace = this->IC.stack_area - this->IC.blocks_area;
450 
451  // rescale terminal pins' locations
452  this->scaleTerminalPins(outline);
453  }
454 
455  return outline;
456  }

460  inline void scaleTerminalPins(Point outline) {
461  double pins_scale_x, pins_scale_y;
462 
463  // scale terminal pins; first determine original pins outline
464  pins_scale_x = pins_scale_y = 0.0;
465  for (Pin const& pin : this->terminals) {
466  pins_scale_x = std::max(pins_scale_x, pin.bb.ll.x);
467  pins_scale_y = std::max(pins_scale_y, pin.bb.ll.y);
468  }
469  // scale terminal pins; scale pin coordinates according to given outline
470  pins_scale_x = outline.x / pins_scale_x;
471  pins_scale_y = outline.y / pins_scale_y;
472  for (Pin& pin : this->terminals) {
473  pin.bb.ll.x *= pins_scale_x;
474  pin.bb.ll.y *= pins_scale_y;
475  // also set upper right to same coordinates, thus pins are ``point''
476  // blocks w/ zero area
477  pin.bb.ur.x = pin.bb.ll.x;
478  pin.bb.ur.y = pin.bb.ll.y;
479  }
480  }
