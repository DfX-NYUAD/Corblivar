490 inline bool ContiguityAnalysis::boundaries_vert_comp(ContiguityAnalysis::Boundary const& b1, ContiguityAnalysis::Boundary const& b2) {
491  return (
492  // x-coordinates are the first criterion; note
493  // that it's sufficient to compare the first
494  // (lower) points since it's a vertical segment
495  (b1.low.x < b2.low.x)
496  // for boundaries with same x-coordinate, resolve
497  // equal values by considering the boundaries'
498  // y-coordinate
499  || (Math::doubleComp(b1.low.x, b2.low.x) && (b1.low.y < b2.low.y))
500  // there will also be cases when segments start on
501  // the same x- and y-coordinate; then, resolve by
502  // block order, i.e., the segment of the left
503  // block comes first
504  || (Math::doubleComp(b1.low.x, b2.low.x) && Math::doubleComp(b1.low.y, b2.low.y) && (b1.block->bb.ll.x < b2.block->bb.ll.x))
505  );
506 };

509 inline bool ContiguityAnalysis::boundaries_hor_comp(ContiguityAnalysis::Boundary const& b1, ContiguityAnalysis::Boundary const& b2) {
510  return (
511  // y-coordinates are the first criterion; note
512  // that it's sufficient to compare the first
513  // (left) points since it's a horizontal segment
514  (b1.low.y < b2.low.y)
515  // for boundaries with same y-coordinate, resolve
516  // equal values by considering the boundaries'
517  // x-coordinate
518  || (Math::doubleComp(b1.low.y, b2.low.y) && (b1.low.x < b2.low.x))
519  // there will also be cases when segments start on
520  // the same y- and x-coordinate; then, resolve by
521  // block order, i.e., the segment of the lower
522  // block comes first
523  || (Math::doubleComp(b1.low.y, b2.low.y) && Math::doubleComp(b1.low.x, b2.low.x) && (b1.block->bb.ll.y < b2.block->bb.ll.y))
524  );
525 };
