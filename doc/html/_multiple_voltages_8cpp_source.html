29 void MultipleVoltages::determineCompoundModules(int layers, std::vector<Block> const& blocks, ContiguityAnalysis& cont) {
30 
31  this->modules.clear();
32 
33  // consider each block as starting point for a compound module
34  for (Block const& start : blocks) {
35 
36  // init the base compound module, containing only the block itself
37  MultipleVoltages::CompoundModule module;
38 
39  // copy feasible voltages
40  module.feasible_voltages = start.feasible_voltages;
41 
42  // init pointers to blocks
43  module.blocks.push_back(&start);
44 
45  // init power saving, based on feasible voltages and current block; note
46  // that previous values are not defined, thus the regular case to reset
47  // and recalculate power saving over all (here one) blocks is applied
48  module.update_power_saving();
49 
50  // init block ids such that they may encode all blocks' numerical ids;
51  // also account for the offset of one, introduced by Block::DUMMY_NUM_ID
52  module.block_ids.reserve(blocks.size() + 1);
53  for (unsigned b = 0; b < blocks.size() + 1; b++) {
54  module.block_ids.push_back(false);
55  }
56 
57  // also, set the block-ids' flag for the current block
58  module.block_ids[start.numerical_id] = true;
59 
60  // init neighbours; pointers to block's neighbour is sufficient
61  for (auto& neighbour : start.contiguous_neighbours) {
62  module.contiguous_neighbours.insert({neighbour.block->numerical_id, &neighbour});
63  }
64 
65  // init outline and corners for power rings
66  module.outline.reserve(layers);
67  module.corners_powerring.reserve(layers);
68 
69  for (int l = 0; l < layers; l++) {
70 
71  // empty bb
72  module.outline.emplace_back(std::vector<Rect>());
73 
74  // any layer, also not affected layers, may be initialized with
75  // the trivial min number of corners, i.e., 4
76  module.corners_powerring.emplace_back(4);
77 
78  if (start.layer == l) {
79  module.outline[l].emplace_back(start.bb);
80  }
81  // note that outline[l] shall remain empty otherwise
82  }
83 
84  // store base compound module
85  auto inserted_it = this->modules.insert(this->modules.begin(), {module.block_ids, std::move(module)});
86 
87  // perform stepwise and recursive merging of base module into larger
88  // compound modules
89  this->buildCompoundModulesHelper(inserted_it->second, inserted_it, cont);
90  }
91 
92  if (MultipleVoltages::DBG) {
93 
94  std::cout << "DBG_VOLTAGES> Compound modules (in total " << this->modules.size() << "):" << std::endl;
95 
96  for (auto it = this->modules.begin(); it != this->modules.end(); ++it) {
97 
98  CompoundModule& module = it->second;
99 
100  std::cout << "DBG_VOLTAGES> Module;" << std::endl;
101  std::cout << "DBG_VOLTAGES> Comprised blocks #: " << module.blocks.size() << std::endl;
102  std::cout << "DBG_VOLTAGES> Comprised blocks ids: " << module.id() << std::endl;
103  std::cout << "DBG_VOLTAGES> Module voltages bitset: " << module.feasible_voltages << std::endl;
104  std::cout << "DBG_VOLTAGES> Index of min voltage: " << module.min_voltage_index() << std::endl;
105  std::cout << "DBG_VOLTAGES> Module (local) cost: " << module.outline_cost << std::endl;
106  }
107  std::cout << "DBG_VOLTAGES>" << std::endl;
108  }

126  // comparator for local multiset of selected compound modules; multiset since cost
127  // may be equal for some modules, especially for trivial modules comprising one
128  // block; the comparator can be parametrized for proper cost normalization
129  //
130  class modules_cost_comp {
131 
132  double max_power_saving;
133  unsigned max_corners;
134  MultipleVoltages::Parameters parameters;
135 
136  public:
137  // initializer for comparator w/ parameters
138  modules_cost_comp(double const& max_power_saving, unsigned const& max_corners, MultipleVoltages::Parameters const& parameters) {
139  this->max_power_saving = max_power_saving;
140  this->max_corners = max_corners;
141  this->parameters = parameters;
142  }
143 
144  bool operator()(CompoundModule const* m1, CompoundModule const* m2) const {
145 
146  double c1 = m1->cost(this->max_power_saving, this->max_corners, this->parameters);
147  double c2 = m2->cost(this->max_power_saving, this->max_corners, this->parameters);
148 
149  return (
150  // the smaller the cost, the better
151  (c1 < c2)
152  // if cost are similar, consider larger modules in
153  // the sense of modules covering more blocks; this
154  // is especially relevant to discourage trivial
155  // modules comprising only one block; these blocks
156  // will have same cost
157  || (Math::doubleComp(c1, c2) && m1->blocks.size() > m2->blocks.size())
158  );
159  }
160  };
161 
162  // first, determine max values for corners and power saving, required for
163  // normalization of related cost terms
164  //
165  max_power_saving = this->modules.begin()->second.power_saving();
166  max_corners = this->modules.begin()->second.corners_powerring_max();
167  for (auto it = this->modules.begin(); it != this->modules.end(); ++it) {
168  max_power_saving = std::max(max_power_saving, it->second.power_saving());
169  max_corners = std::max(max_corners, it->second.corners_powerring_max());
170  }
171 
172  // now, a multiset can be initialized with the proper comparator parameters
173  //
174  std::multiset<CompoundModule*, modules_cost_comp> modules_w_cost (modules_cost_comp(max_power_saving, max_corners, this->parameters));
175 
176  // second, insert all modules' pointers into this multiset
177  //
178  for (auto it = this->modules.begin(); it != this->modules.end(); ++it) {
179  modules_w_cost.insert(&(it->second));
180  }
181 
182  // third, stepwise select module with best cost, assign module's voltage to all
183  // related modules, remove the other (candidate) modules which comprise any of the
184  // already assigned blocks (to avoid redundant assignments with non-optimal cost
185  // for any block); proceed until all modules have been considered, which implies
186  // until all blocks have a cost-optimal voltage assignment
187  //
188  this->selected_modules.clear();
189  while (!modules_w_cost.empty()) {
190 
191  if (MultipleVoltages::DBG_VERBOSE) {
192 
193  std::cout << "DBG_VOLTAGES> Current set of compound modules to be considered (in total " << modules_w_cost.size() << "); view ordered by total cost:" << std::endl;
194 
195  for (auto* module : modules_w_cost) {
196 
197  std::cout << "DBG_VOLTAGES> Module;" << std::endl;
198  std::cout << "DBG_VOLTAGES> Comprised blocks #: " << module->blocks.size() << std::endl;
199  std::cout << "DBG_VOLTAGES> Comprised blocks ids: " << module->id() << std::endl;
200  std::cout << "DBG_VOLTAGES> Module voltages bitset: " << module->feasible_voltages << std::endl;
201  std::cout << "DBG_VOLTAGES> Index of min voltage: " << module->min_voltage_index() << std::endl;
202  std::cout << "DBG_VOLTAGES> Module (total) cost: " << module->cost(max_power_saving, max_corners, parameters) << std::endl;
203  std::cout << "DBG_VOLTAGES> Gain minus ``wasted gain'' in power reduction: " << module->power_saving() << std::endl;
204  std::cout << "DBG_VOLTAGES> Gain in power reduction: " << module->power_saving(false) << std::endl;
205  std::cout << "DBG_VOLTAGES> Estimated max number of corners for power rings: " << module->corners_powerring_max() << std::endl;
206  std::cout << "DBG_VOLTAGES> Covered blocks (not modeled in cost, but considered during selection): " << module->blocks.size() << std::endl;
207 
208  }
209  std::cout << "DBG_VOLTAGES>" << std::endl;
210  }
211 
212  // select module with currently best cost
213  cur_selected_module = *(modules_w_cost.begin());
214 
215  // memorize this module as selected
216  this->selected_modules.push_back(cur_selected_module);
217 
218  // assign related values to all blocks comprised in this module: (index
219  // of) lowest applicable voltage, and pointer to module itself
220  //
221  min_voltage_index = cur_selected_module->min_voltage_index();
222  for (Block const* b : cur_selected_module->blocks) {
223 
224  b->assigned_voltage_index = min_voltage_index;
225  b->assigned_module = cur_selected_module;
226  }
227 
228  if (MultipleVoltages::DBG_VERBOSE) {
229 
230  std::cout << "DBG_VOLTAGES> Selected compound module (out of " << modules_w_cost.size() << " modules);" << std::endl;
231  std::cout << "DBG_VOLTAGES> Comprised blocks #: " << cur_selected_module->blocks.size() << std::endl;
232  std::cout << "DBG_VOLTAGES> Comprised blocks ids: " << cur_selected_module->id() << std::endl;
233  std::cout << "DBG_VOLTAGES> Module voltages bitset: " << cur_selected_module->feasible_voltages << std::endl;
234  std::cout << "DBG_VOLTAGES> Index of min voltage: " << min_voltage_index << std::endl;
235  std::cout << "DBG_VOLTAGES> Module (total) cost: " << cur_selected_module->cost(max_power_saving, max_corners, parameters) << std::endl;
236  std::cout << "DBG_VOLTAGES> Gain minus ``wasted gain'' in power reduction: " << cur_selected_module->power_saving() << std::endl;
237  std::cout << "DBG_VOLTAGES> Gain in power reduction: " << cur_selected_module->power_saving(false) << std::endl;
238  std::cout << "DBG_VOLTAGES> Estimated max number of corners for power rings: " << cur_selected_module->corners_powerring_max() << std::endl;
239  std::cout << "DBG_VOLTAGES> Covered blocks (not modeled in cost, but considered during selection): " << cur_selected_module->blocks.size() << std::endl;
240  }
241 
242  // remove other modules which contain some already assigned blocks; start
243  // with 1st module in set to also remove the just considered module
244  //
245  if (MultipleVoltages::DBG_VERBOSE) {
246  count = 0;
247  }
248 
249  for (auto it = modules_w_cost.begin(); it != modules_w_cost.end();) {
250 
251  module_to_check = *it;
252  module_to_remove = false;
253 
254  for (Block const* assigned_block : cur_selected_module->blocks) {
255 
256  // the module to check contains a block which is assigned
257  // in the current module; thus, we drop the module
258  //
259  if (module_to_check->block_ids[assigned_block->numerical_id] == true) {
260 
261  if (MultipleVoltages::DBG_VERBOSE) {
262 
263  count++;
264 
265  std::cout << "DBG_VOLTAGES> Module to be deleted after selecting the module above: " << module_to_check->id() << std::endl;
266  }
267 
268  // also update iterator; pointing to next element
269  // after erased element
270  it = modules_w_cost.erase(it);
271  module_to_remove = true;
272 
273  break;
274  }
275  }
276 
277  // no module to remove; simply increment iterator
278  if (!module_to_remove) {
279  ++it;
280  }
281  }
282 
283  if (MultipleVoltages::DBG_VERBOSE) {
284  std::cout << "DBG_VOLTAGES> Deleted modules count: " << count << std::endl;
285  }
286  }
287 
288  // fourth, merge selected modules whenever possible, i.e., when some of the
289  // modules' blocks are contiguous to another module sharing the same voltage
290  //
291  // note that such merging will a) impact the corners and b) undermine the cost
292  // normalization and thus the actual top-down selection. For a), it most likely
293  // increases the estimated max number of corners to an unreasonably high value
294  // since most merges will not increase the numbers of corners, especially not when
295  // many modules nearby (with same voltages) are merged. For b), the actual cost of
296  // merged modules shall be different and may be inferior than other cost,
297  // suggesting that the merge is not beneficial
298  //
299  // thus, it shall only be applied when requested, e.g., for final logging
300  if (merge_selected_modules) {
301 
302  if (MultipleVoltages::DBG) {
303  std::cout << "DBG_VOLTAGES> Start merging modules" << std::endl;
304  }
305 
306  // for-loop instead of iterator, since we edit this very data structure
307  //
308  for (unsigned m = 0; m < this->selected_modules.size(); m++) {
309 
310  MultipleVoltages::CompoundModule* module = this->selected_modules[m];
311 
312  // walk all contiguous blocks of current module
313  //
314  for (auto it = module->contiguous_neighbours.begin(); it != module->contiguous_neighbours.end(); ++it) {
315 
316  MultipleVoltages::CompoundModule* n_module = it->second->block->assigned_module;
317 
318  // if the related module of the contiguous block has the same
319  // voltage index as this module, they can be merged; merging means
320  // to extend the current module with the blocks from the
321  // contiguous block's module
322  //
323  if (n_module->min_voltage_index() == module->min_voltage_index()) {
324 
325  // sanity check; avoid merging with itself
326  if (n_module->block_ids == module->block_ids) {
327  continue;
328  }
329 
330  if (MultipleVoltages::DBG) {
331  std::cout << "DBG_VOLTAGES> Merging modules;" << std::endl;
332  std::cout << "DBG_VOLTAGES> " << module->id() << std::endl;
333  std::cout << "DBG_VOLTAGES> " << n_module->id() << std::endl;
334  }
335 
336  // update the actual blocks
337  for (Block const* b : n_module->blocks) {
338 
339  module->blocks.push_back(b);
340  module->block_ids[b->numerical_id] = true;
341 
342  // also update the module pointer for merged
343  // module's blocks
344  b->assigned_module = module;
345  }
346 
347  // sum up the power saving values
348  module->power_saving_ += n_module->power_saving_;
349  module->power_saving_wasted_ += n_module->power_saving_wasted_;
350 
351  // add the outline rects from the module to be merged
352  for (unsigned l = 0; l < module->outline.size(); l++) {
353 
354  for (auto& rect : n_module->outline[l]) {
355  module->outline[l].push_back(std::move(rect));
356  }
357 
358  // also sum up the power-ring corners, but
359  // subtract two under the assumption that the
360  // previous module's outline can be extended
361  // without further corners; this is an somewhat
362  // optimistic but simple estimation
363  module->corners_powerring[l] += n_module->corners_powerring[l] - 2;
364  }
365 
366  // add (pointers to) now additionally considered
367  // contiguous neighbours; note that only yet not
368  // considered neighbours are effectively added
369  for (auto n : n_module->contiguous_neighbours) {
370 
371  // ignore any neighbour which is already comprised in the module
372  if (module->block_ids[n.first] == true) {
373  continue;
374  }
375 
376  module->contiguous_neighbours.insert(n);
377  }
378 
379  // erase the just merged module
380  //
381  for (auto it = this->selected_modules.begin(); it != this->selected_modules.end(); ++it) {
382 
383  if ((*it)->block_ids == n_module->block_ids) {
384  this->selected_modules.erase(it);
385  break;
386  }
387  }
388 
389  // finally, reset the iterator for contiguous blocks as
390  // well; required to capture transitive merges and,
391  // furthermore, iterator it is invalid after the above
392  // insertions
393  //
394  it = module->contiguous_neighbours.begin();
395  }
396  }
397  }
398 
399  if (MultipleVoltages::DBG) {
400  std::cout << "DBG_VOLTAGES> Done merging modules" << std::endl;
401  std::cout << "DBG_VOLTAGES>" << std::endl;
402  }
403  }
404 
405  if (MultipleVoltages::DBG) {
406 
407  count = 0;
408 
409  std::cout << "DBG_VOLTAGES> Selected compound modules (in total " << this->selected_modules.size() << "); view ordered by total cost:" << std::endl;
410 
411  for (auto* module : this->selected_modules) {
412 
413  std::cout << "DBG_VOLTAGES> Module;" << std::endl;
414  std::cout << "DBG_VOLTAGES> Comprised blocks #: " << module->blocks.size() << std::endl;
415  std::cout << "DBG_VOLTAGES> Comprised blocks ids: " << module->id() << std::endl;
416  std::cout << "DBG_VOLTAGES> Module voltages bitset: " << module->feasible_voltages << std::endl;
417  std::cout << "DBG_VOLTAGES> Index of min voltage: " << module->min_voltage_index() << std::endl;
418  std::cout << "DBG_VOLTAGES> Module (total) cost: " << module->cost(max_power_saving, max_corners, parameters) << std::endl;
419  std::cout << "DBG_VOLTAGES> Gain minus ``wasted gain'' in power reduction: " << module->power_saving() << std::endl;
420  std::cout << "DBG_VOLTAGES> Gain in power reduction: " << module->power_saving(false) << std::endl;
421  std::cout << "DBG_VOLTAGES> Estimated max number of corners for power rings: " << module->corners_powerring_max() << std::endl;
422  std::cout << "DBG_VOLTAGES> Covered blocks (not modeled in cost, but considered during selection): " << module->blocks.size() << std::endl;
423 
424  count += module->blocks.size();
425  }
426  std::cout << "DBG_VOLTAGES>" << std::endl;
427  std::cout << "DBG_VOLTAGES> In total assigned blocks to modules: " << count << std::endl;
428  std::cout << "DBG_VOLTAGES>" << std::endl;
429  }
430 
431  return this->selected_modules;

439 void MultipleVoltages::buildCompoundModulesHelper(MultipleVoltages::CompoundModule& module, MultipleVoltages::modules_type::iterator hint, ContiguityAnalysis& cont) {
440  std::bitset<MultipleVoltages::MAX_VOLTAGES> feasible_voltages;
441  ContiguityAnalysis::ContiguousNeighbour* neighbour;
442  std::vector<ContiguityAnalysis::ContiguousNeighbour*> candidates;
443  double best_candidate_cost, cur_candidate_cost;
444  ContiguityAnalysis::ContiguousNeighbour* best_candidate;
445 
446  // walk all current neighbours; perform breadth-first search for each next-level
447  // compound module with same set of applicable voltages
448  //
449  for (auto it = module.contiguous_neighbours.begin(); it != module.contiguous_neighbours.end(); ++it) {
450 
451  neighbour = it->second;
452 
453  // first, we determine if adding this neighbour would lead to an trivial
454  // solution, i.e., only the highest possible voltage is assignable; such
455  // modules are mainly ignored (one exception, see below) and thus we can
456  // achieve notable reduction in memory and runtime by pruning trivial
457  // solutions early on during recursive bottom-up phase
458  //
459  // the only exception where modules with only highest voltage shall be
460  // further investigated is in case adjacent trivial compound modules
461  // (single blocks) can be merged
462  //
463 
464  // bit-wise AND to obtain the intersection of feasible voltages
465  feasible_voltages = module.feasible_voltages & neighbour->block->feasible_voltages;
466 
467  if (MultipleVoltages::DBG) {
468 
469  std::cout << "DBG_VOLTAGES> Current module (" << module.id() << "),(" << module.feasible_voltages << ");";
470  std::cout << " consider neighbour block: (" << neighbour->block->id << "),(" << neighbour->block->feasible_voltages << ")" << std::endl;
471  }
472 
473  // more than one voltage is applicable _afterwards_ but the resulting set
474  // of voltages is the same as _before_ for the previous module
475  //
476  // here, we don't insert the new module immediately, but rather memorize
477  // all such candidate modules / neighbours and then consider only the one
478  // with the lowest cost for further branching
479  if (feasible_voltages.count() > 1 && feasible_voltages == module.feasible_voltages) {
480 
481  if (MultipleVoltages::DBG) {
482  std::cout << "DBG_VOLTAGES> No change in applicable voltages (" << module.feasible_voltages << ")";
483  std::cout << "; consider neighbour block as candidate" << std::endl;
484  }
485 
486  candidates.push_back(neighbour);
487  }
488  // only one voltage was applicable, i.e., handle a trivial compound
489  // module; consider only for merging with another trivial block/neighbour;
490  // this way, largest possible islands for the trivial voltage can be
491  // obtained; in order to limit the search space, branching is not allowed
492  // here, i.e., modules are stepwise added as long as some contiguous and
493  // trivial modules are available, but once such a module is selected for
494  // merging, no further modules on this branching level are considered
495  //
496  // same candidate principle as for other cases with unchanged
497  // set of voltages applies here, in order to limit the search space
498  else if (module.feasible_voltages.count() == 1 && neighbour->block->feasible_voltages.count() == 1) {
499 
500  if (MultipleVoltages::DBG) {
501  std::cout << "DBG_VOLTAGES> Consider trivial module to merge with another trivial block/neighbour;";
502  std::cout << " consider neighbour block as candidate" << std::endl;
503  }
504 
505  // previous neighbours shall not be considered, in order to limit
506  // the search space such that only ``forward merging'' of new
507  // contiguous trivial modules is considered
508  this->insertCompoundModuleHelper(module, neighbour, false, feasible_voltages, hint, cont);
509 
510  // this break is the ``trick'' for disabling branching: once a
511  // contiguous trivial module is extended by this relevant
512  // neighbour and once the recursive calls (for building up
513  // resulting larger modules) return to this point, no further
514  // neighbours are considered; the resulting stepwise merging of
515  // only one trivial neighbour is sufficient to capture
516  // largest-possible contiguous trivial modules
517  break;
518  }
519  // more than one voltage is applicable, and the set of voltages has
520  // changed; such a module should be considered without notice of cost,
521  // since it impacts the overall set of possible voltage islands
522  //
523  else if (feasible_voltages.count() > 1 && feasible_voltages != module.feasible_voltages) {
524 
525  if (MultipleVoltages::DBG) {
526  std::cout << "DBG_VOLTAGES> Change in applicable voltages: " << module.feasible_voltages << " before, " << feasible_voltages << " now;";
527  std::cout << " non-trivial solution; try insertion of related new module" << std::endl;
528  }
529 
530  // previous neighbours shall be considered, since the related new
531  // module has a different set of voltages, i.e., no tie-braking
532  // was considered among some candidate neighbours
533  this->insertCompoundModuleHelper(module, neighbour, true, feasible_voltages, hint, cont);
534  }
535  // any other case, i.e., only one (trivially the highest possible) voltage
536  // applicable for the new module; to be ignored
537  else {
538  if (MultipleVoltages::DBG) {
539  std::cout << "DBG_VOLTAGES> Trivial partial solution, with only highest voltage applicable (" << feasible_voltages << ");";
540  std::cout << " skip this neighbour block" << std::endl;
541  }
542 
543  continue;
544  }
545  }
546 
547  if (MultipleVoltages::DBG) {
548  std::cout << "DBG_VOLTAGES> Current module (" << module.id() << "),(" << module.feasible_voltages << "); all neighbour blocks considered" << std::endl;
549  }
550 
551  // some neighbours may be added such that there is no change in the set of
552  // applicable voltages; out of the related candidates, proceed only with the
553  // lowest-cost candidate (w.r.t. local outline_cost); this way, the solution space
554  // is notably reduced, and the top-down process would select compound modules of
555  // lowest cost (global cost, somewhat related to this local cost) anyway, thus
556  // this decision can already be applied here
557  //
558  if (!candidates.empty()) {
559 
560  if (MultipleVoltages::DBG) {
561  std::cout << "DBG_VOLTAGES> Current module (" << module.id() << "),(" << module.feasible_voltages << "); evaluate candidates" << std::endl;
562  }
563 
564  // init with dummy cost (each bb cannot be more intruded than by factor
565  // 1.0); min cost is to be determined
566  best_candidate_cost = 1.0;
567 
568  // determine best candidate
569  for (auto* candidate : candidates) {
570 
571  // apply_update = false; i.e., only calculate cost of potentially
572  // adding the candidate block, don't add block yet
573  //
574  cur_candidate_cost = module.updateOutlineCost(candidate, cont, false);
575 
576  if (MultipleVoltages::DBG) {
577  std::cout << "DBG_VOLTAGES> Candidate block " << candidate->block->id <<"; cost: " << cur_candidate_cost << std::endl;
578  }
579 
580  // determine min cost and related best candidate
581  if (cur_candidate_cost < best_candidate_cost) {
582  best_candidate_cost = cur_candidate_cost;
583  best_candidate = candidate;
584  }
585  }
586 
587  // redetermine intersection of feasible voltages for best-cost candidate
588  feasible_voltages = module.feasible_voltages & best_candidate->block->feasible_voltages;
589 
590  if (MultipleVoltages::DBG) {
591  std::cout << "DBG_VOLTAGES> Current module (" << module.id() << "),(" << module.feasible_voltages << ");";
592  std::cout << " best candidate block " << best_candidate->block->id;
593  std::cout << "; cost: " << best_candidate_cost << "; try insertion of related new module" << std::endl;
594  }
595 
596  // merge only the best-cost candidate into the module; continue
597  // recursively with this new module; other neighbours shall not be
598  // considered anymore, otherwise the selection of best-cost candidate
599  // would be undermined; note that in practice some blocks will still be
600  // (rightfully) considered since they are also contiguous neighbours with
601  // the now considered best-cost candidate
602  this->insertCompoundModuleHelper(module, best_candidate, false, feasible_voltages, hint, cont);
603  }
604 }

606 inline void MultipleVoltages::insertCompoundModuleHelper(MultipleVoltages::CompoundModule& module, ContiguityAnalysis::ContiguousNeighbour* neighbour, bool consider_prev_neighbours, std::bitset<MultipleVoltages::MAX_VOLTAGES>& feasible_voltages, MultipleVoltages::modules_type::iterator& hint, ContiguityAnalysis& cont) {
607  MultipleVoltages::modules_type::iterator inserted;
608 
609  // first, we have to check whether this potential compound module was already
610  // considered previously, i.e., during consideration of another starting module;
611  // only if the compound module is really a new one, then we continue
612  //
613  // to check if the potential module already exits, we ``leverage'' the previous
614  // module, by assigning the now-to-consider neighbour's block to the previous
615  // module's set of considered blocks; thus we avoid copying the whole set of
616  // blocks just for checking the potential module's existence; for further improved
617  // efficiency, we leverage the bool flags instead of the actual set blocks
618  //
619  module.block_ids[neighbour->block->numerical_id] = true;
620 
621  // now, perform the actual check
622  if (this->modules.find(module.block_ids) != this->modules.end()) {
623 
624  // the potential module does already exit; revert the just assigned
625  // neighbour from the previous module again; and return
626  module.block_ids[neighbour->block->numerical_id] = false;
627 
628  if (MultipleVoltages::DBG) {
629  std::cout << "DBG_VOLTAGES> Insertion not successful; module was already inserted previously" << std::endl;
630  }
631 
632  return;
633  }
634 
635  // at this point, it's clear that we have to generate the new compound module; it
636  // comprises the previous module and the neighbour
637  //
638  MultipleVoltages::CompoundModule new_module;
639 
640  // the blocks assignment is contained in the previous module, since the
641  // neighbour's block was already assigned; simply copy these flags
642  new_module.block_ids = module.block_ids;
643 
644  // only now we shall revert the neighbour's block assignment to the previous module
645  module.block_ids[neighbour->block->numerical_id] = false;
646 
647  // copy block pointers from previous module
648  new_module.blocks = module.blocks;
649  // consider neighbour block for new module
650  new_module.blocks.push_back(neighbour->block);
651 
652  // assign feasible voltages, may be moved since they were generated only for this
653  // callee and will not be reused in the caller
654  new_module.feasible_voltages = std::move(feasible_voltages);
655 
656  // recalculate the power saving for all comprised modules if required, i.e.,
657  // whenever the set of applicable voltages changes
658  //
659  if (new_module.feasible_voltages != module.feasible_voltages) {
660 
661  new_module.update_power_saving();
662  }
663  // otherwise, copy previous values and update only according to the specific new
664  // block to be considered
665  //
666  else {
667  // copy previous values
668  new_module.power_saving_ = module.power_saving_;
669  new_module.power_saving_wasted_ = module.power_saving_wasted_;
670 
671  // update copied values to consider new block
672  new_module.update_power_saving(neighbour->block);
673  }
674 
675  // copy outline from the previous module
676  new_module.outline = module.outline;
677 
678  // copy corners from the previous module
679  new_module.corners_powerring = module.corners_powerring;
680 
681  // update bounding box, blocks area, and recalculate outline cost; all w.r.t.
682  // added (neighbour) block
683  new_module.updateOutlineCost(neighbour, cont);
684 
685  // if previous neighbours shall be considered, copy the related pointers from the
686  // previous module
687  if (consider_prev_neighbours) {
688 
689  new_module.contiguous_neighbours = module.contiguous_neighbours;
690 
691  // we have to ignore the just considered neighbour; deleting afterwards is
692  // computationally less expansive than checking each neighbor's id during
693  // copying
694  new_module.contiguous_neighbours.erase(neighbour->block->numerical_id);
695  }
696 
697  // add (pointers to) neighbours of the now additionally considered block; note
698  // that only yet not considered neighbours are effectively added
699  //
700  for (auto& n : neighbour->block->contiguous_neighbours) {
701 
702  // we have to ignore any neighbour which is already comprised in the
703  // module itself
704  if (module.block_ids[n.block->numerical_id] == true) {
705  continue;
706  }
707  // also, for reasons of reducing computational complexity, no transitively
708  // contiguous neighbours across more than two adjacent dies are considered
709  //
710  // note that all possible inter-die contiguous neighbours between any pair
711  // of adjacent dies were previously determined during
712  // ContiguityAnalysis::analyseBlocks, and by allowing new neighbours to be
713  // within the same die, all possible combinations of transitively
714  // contiguous modules among adjacent dies are still captured, i.e., final
715  // modules may still span more than two dies
716  if (n.block->layer != neighbour->block->layer) {
717  continue;
718  }
719 
720  new_module.contiguous_neighbours.insert({n.block->numerical_id, &n});
721  }
722 
723 
724  // perform actual insertion; hint provided is the iterator to the previously
725  // inserted module
726  //
727  inserted = this->modules.insert(hint, {new_module.block_ids, std::move(new_module)});
728 
729  if (MultipleVoltages::DBG) {
730  std::cout << "DBG_VOLTAGES> Insertion successful; continue recursively with this module" << std::endl;
731  }
732 
733  // recursive call; provide iterator to just inserted new module as hint for next
734  // insertion
735  this->buildCompoundModulesHelper(inserted->second, inserted, cont);
736 }

751 double MultipleVoltages::CompoundModule::updateOutlineCost(ContiguityAnalysis::ContiguousNeighbour* neighbour, ContiguityAnalysis& cont, bool apply_update) {
752  double cost;
753  int n_l = neighbour->block->layer;
754  double intrusion_area = 0.0;
755  bool checked_boundaries = false;
756  Rect ext_bb;
757  Rect neighbour_ext_bb;
758  Rect prev_bb_ext;
759  std::vector<Block const*> intruding_blocks;
760 
761  if (MultipleVoltages::DBG) {
762  if (apply_update) {
763  std::cout << "DBG_VOLTAGES> Update outline cost and power-ring corners; module " << this->id() << ";";
764  }
765  else {
766  std::cout << "DBG_VOLTAGES> Determine (but don't update) outline cost and power-ring corners; module " << this->id() << ";";
767  }
768  std::cout << " neighbour block " << neighbour->block->id << "; affected die " << n_l << std::endl;
769  }
770 
771  // update bounding boxes on (by added block) affected die; note that the added
772  // block may be the first on its related die which is assigned to this module;
773  // then init new bb
774  //
775  if (this->outline[n_l].empty()) {
776 
777  // apply update only when required
778  if (apply_update) {
779  this->outline[n_l].emplace_back(neighbour->block->bb);
780  }
781 
782  // power-ring corners can safely be ignored; adding one rectangular block
783  // will not increase the previous max value for power-ring corners
784 
785  // the first module's bb will not be intruded per se
786  cost = 0.0;
787  }
788  // update existing bb; try to extend bb to cover previous blocks and the new
789  // neighbour block; check for intrusion by any other block
790  //
791  else {
792  // consider the previous bb to be extended by the neighbour; the relevant
793  // bb is the last one of the vector (by this just introduced definition);
794  // consider local copy and only store in case update shall be applied
795  //
796  ext_bb = Rect::determBoundingBox(this->outline[n_l].back(), neighbour->block->bb);
797 
798  if (MultipleVoltages::DBG) {
799  std::cout << "DBG_VOLTAGES> Currently considered extended bb ";
800  std::cout << "(" << ext_bb.ll.x << "," << ext_bb.ll.y << ")";
801  std::cout << "(" << ext_bb.ur.x << "," << ext_bb.ur.y << ")" << std::endl;
802  }
803 
804  // walking vertical boundaries, provided by ContiguityAnalysis; the
805  // boundaries of the relevant die will be checked whether they intrude the
806  // extended bb, and if so to what degree; note that walking the vertical
807  // boundaries is sufficient for determining overlaps in x- and
808  // y-dimension; also see ContiguityAnalysis::analyseBlocks
809  for (auto i1 = cont.boundaries_vert[n_l].begin(); i1 != cont.boundaries_vert[n_l].end(); ++i1) {
810 
811  ContiguityAnalysis::Boundary const& b1 = (*i1);
812 
813  // the boundary b2, to be compared to b1, should be within the
814  // x-range of the extended bb; thus we initially search for the
815  // first boundary (slightly larger than) the extended bb's left
816  // x-coordinate
817  if (b1.low.x <= ext_bb.ll.x) {
818  continue;
819  }
820 
821  // at this point, a boundary b1 is found which is greater than the
822  // lower x-coordinate of the extended bb; check for intruding
823  // boundaries/blocks
824  for (auto i2 = i1; i2 != cont.boundaries_vert[n_l].end(); ++i2) {
825 
826  ContiguityAnalysis::Boundary const& b2 = (*i2);
827 
828  // break condition; if b2 is just touching (or later on
829  // outside to) the right of extended bb, no intersection
830  // if feasible anymore
831  if (b2.low.x >= ext_bb.ur.x) {
832 
833  checked_boundaries = true;
834  break;
835  }
836 
837  // some intersection _may_ exist, but only for a) blocks
838  // not covered in the module yet or not being the
839  // neighbour and b) if there is some overlap in
840  // y-direction
841  //
842  // negation of a), ignore such block
843  if (b2.block->numerical_id == neighbour->block->numerical_id) {
844  continue;
845  }
846  if (this->block_ids[b2.block->numerical_id] == true) {
847  continue;
848  }
849  // b)
850  if (ext_bb.ll.y < b2.high.y && b2.low.y < ext_bb.ur.y) {
851 
852  // at this point, we know that b2 is intersecting
853  // with extended bb to some degree in _both_
854  // dimensions; we may memorize the _potentially_
855  // intruding block (in a map to avoid considering
856  // blocks two times which may happen when walking
857  // the two vertical boundaries of all blocks)
858  //
859  // finally, look ahead whether this blocks
860  // represents an relevant intrusion, i.e., whether
861  // the voltages will be different; this can only
862  // be addressed conservatively, since the actual
863  // assignment is not done yet: when the intruding
864  // block has a different set of voltages
865  // applicable than the module's current set of
866  // voltage we shall assume this block to be
867  // intruding at this point, since such neighbours,
868  // if merged into the module, will be altering the
869  // set of applicable voltages and thus change the
870  // module's properties altogether, or, if not
871  // merged, will be intruding the module; note that
872  // this consideration will always result in
873  // trivial neighbours (with only highest voltage
874  // applicable) to be rightfully considered as
875  // intruding; such modules will not be generated
876  // where trivial neighbours are merged into
877  //
878  if (this->feasible_voltages != b2.block->feasible_voltages) {
879  intruding_blocks.push_back(b2.block);
880  }
881  }
882  }
883 
884  // break condition, all relevant boundaries have been considered
885  if (checked_boundaries) {
886  break;
887  }
888  }
889 
890  // in case no intrusion would occur, consider the extended bb
891  if (intruding_blocks.empty()) {
892 
893  if (apply_update) {
894  this->outline[n_l].back() = ext_bb;
895  }
896 
897  if (MultipleVoltages::DBG) {
898  std::cout << "DBG_VOLTAGES> Extended bb is not intruded by any block; consider this extended bb as is" << std::endl;
899  }
900 
901  // note that no increase in corners for the power rings occurs in
902  // such cases; thus they are ignored
903  }
904  // in case any intrusion would occur, consider only the separate,
905  // non-intruded boxes
906  //
907  // also handle the estimated number of corners in the power rings
908  //
909  else {
910  // consider each intruding block only once, i.e., sort and make
911  // vector unique; for efficiency, a vector which is later on
912  // sorted and made unique is usually still better than using a
913  // sorted set or a hashed map
914  //
915  std::sort(intruding_blocks.begin(), intruding_blocks.end(),
916  // lambda expression for sorting
917  [&](Block const* b1, Block const* b2) {
918  return b1->numerical_id < b2->numerical_id;
919  }
920  );
921  auto it_last = std::unique(intruding_blocks.begin(), intruding_blocks.end(),
922  // lambda expression for equality
923  [&](Block const* b1, Block const* b2) {
924  return b1->numerical_id == b2->numerical_id;
925  }
926  );
927  // note that the vector must be resized, otherwise its range is
928  // not valid in the sense that it may still refer to some
929  // redundant entries
930  intruding_blocks.resize(std::distance(intruding_blocks.begin(), it_last));
931 
932  // add the neighbours (extended) bb and extend the previous bb
933  // separately; the extension shall be applied such that number of
934  // corners will be minimized, i.e., the bbs should be sized to
935  // match the overall bb (enclosing previous bb and neighbour) as
936  // close as possible but still considering intruding blocks
937  //
938  // init copy for extended neighbour bb with actual neighbour bb
939  neighbour_ext_bb = neighbour->block->bb;
940  // hold copy of actual prev bb; the last in the container by
941  // definition
942  Rect prev_bb = this->outline[n_l].back();
943  // init copy for extended prev bb with actual prev bb
944  prev_bb_ext = prev_bb;
945 
946  // extent both bbs to meet boundaries of overall bb; to do so,
947  // increase the bbs separately in the relevant dimensions
948  //
949  // prev bb and neighbour are vertically intersecting, thus extend
950  // the vertical dimensions
951  if (Rect::rectsIntersectVertical(neighbour->block->bb, prev_bb)) {
952  neighbour_ext_bb.ll.y = prev_bb_ext.ll.y = std::min(neighbour->block->bb.ll.y, prev_bb.ll.y);
953  neighbour_ext_bb.ur.y = prev_bb_ext.ur.y = std::max(neighbour->block->bb.ur.y, prev_bb.ur.y);
954  }
955  // prev bb and neighbour are horizontally intersecting, thus
956  // extend the horizontal dimensions
957  else if (Rect::rectsIntersectHorizontal(neighbour->block->bb, prev_bb)) {
958  neighbour_ext_bb.ll.x = prev_bb_ext.ll.x = std::min(neighbour->block->bb.ll.x, prev_bb.ll.x);
959  neighbour_ext_bb.ur.x = prev_bb_ext.ur.x = std::max(neighbour->block->bb.ur.x, prev_bb.ur.x);
960  }
961 
962  // determine the amount of intersection/intrusion; and ``cut''
963  // parts of extended bbs which are intruded
964  //
965  for (Block const* intruding_block : intruding_blocks) {
966 
967  Rect const& cur_intruding_bb = intruding_block->bb;
968 
969  // if the intruding block is below the neighbour, consider
970  // to limit the lower boundary of the extended bb
971  //
972  // note that checking for intersection is not required
973  // since neighbours are continuous by definition, the same
974  // applies to the other cases below
975  //
976  if (Rect::rectA_below_rectB(cur_intruding_bb, neighbour->block->bb, false)) {
977  neighbour_ext_bb.ll.y = std::max(cur_intruding_bb.ur.y, neighbour_ext_bb.ll.y);
978  }
979  // if the intruding block is above the neighbour, consider
980  // to limit the upper boundary of the extended bb
981  else if (Rect::rectA_below_rectB(neighbour->block->bb, cur_intruding_bb, false)) {
982  neighbour_ext_bb.ur.y = std::min(cur_intruding_bb.ll.y, neighbour_ext_bb.ur.y);
983  }
984 
985  // if the intruding block is left of the neighbour,
986  // consider to limit the left boundary of the extended bb
987  if (Rect::rectA_leftOf_rectB(cur_intruding_bb, neighbour->block->bb, false)) {
988  neighbour_ext_bb.ll.x = std::max(cur_intruding_bb.ur.x, neighbour_ext_bb.ll.x);
989  }
990  // if the intruding block is right of the neighbour,
991  // consider to limit the right boundary of the extended bb
992  else if (Rect::rectA_leftOf_rectB(neighbour->block->bb, cur_intruding_bb, false)) {
993  neighbour_ext_bb.ur.x = std::min(cur_intruding_bb.ll.x, neighbour_ext_bb.ur.x);
994  }
995 
996  // same as above also applies to the prev bb
997  //
998  // if the intruding block is below the prev bb, consider
999  // to limit the lower boundary of the extended bb
1000  if (Rect::rectA_below_rectB(cur_intruding_bb, prev_bb, false)) {
1001  prev_bb_ext.ll.y = std::max(cur_intruding_bb.ur.y, prev_bb.ll.y);
1002  }
1003  // if the intruding block is above the prev bb, consider
1004  // to limit the upper boundary of the extended bb
1005  else if (Rect::rectA_below_rectB(prev_bb, cur_intruding_bb, false)) {
1006  prev_bb_ext.ur.y = std::min(cur_intruding_bb.ll.y, prev_bb.ur.y);
1007  }
1008 
1009  // if the intruding block is left of the prev bb,
1010  // consider to limit the left boundary of the extended bb
1011  if (Rect::rectA_leftOf_rectB(cur_intruding_bb, prev_bb, false)) {
1012  prev_bb_ext.ll.x = std::max(cur_intruding_bb.ur.x, prev_bb.ll.x);
1013  }
1014  // if the intruding block is right of the prev bb,
1015  // consider to limit the right boundary of the extended bb
1016  else if (Rect::rectA_leftOf_rectB(prev_bb, cur_intruding_bb, false)) {
1017  prev_bb_ext.ur.x = std::min(cur_intruding_bb.ll.x, prev_bb.ur.x);
1018  }
1019 
1020  // determine the amount of intrusion; only consider the
1021  // actual intersection
1022  intrusion_area += Rect::determineIntersection(ext_bb, cur_intruding_bb).area;
1023 
1024  if (MultipleVoltages::DBG) {
1025  std::cout << "DBG_VOLTAGES> Extended bb is intruded by block " << intruding_block->id;
1026  std::cout << "; block bb (" << cur_intruding_bb.ll.x << "," << cur_intruding_bb.ll.y << ")";
1027  std::cout << "(" << cur_intruding_bb.ur.x << "," << cur_intruding_bb.ur.y << ")";
1028  std::cout << "; amount of intrusion / area of intersection: " << Rect::determineIntersection(ext_bb, cur_intruding_bb).area << std::endl;
1029  }
1030  }
1031 
1032  // memorize the extended bbs if required
1033  //
1034  if (apply_update) {
1035 
1036  // recall that prev_bb refers to the previous bb in the
1037  // outline[n_l] by definition; thus, the extended prev bb
1038  // shall replace this very previous bb
1039  this->outline[n_l].back() = prev_bb_ext;
1040 
1041  // store the new, extended bb for the neighbour
1042  this->outline[n_l].emplace_back(neighbour_ext_bb);
1043  }
1044 
1045  // also update the number of corners if required
1046  //
1047  if (apply_update) {
1048  // whenever the extended bbs have different coordinates in
1049  // the extended dimension (due to intruding blocks
1050  // considered above), two new corners will be introduced
1051  //
1052  // prev bb and neighbour are vertically intersecting, thus
1053  // the vertical dimensions were extended
1054  if (Rect::rectsIntersectVertical(neighbour->block->bb, prev_bb)) {
1055 
1056  // check both boundaries separately
1057  if (!Math::doubleComp(neighbour_ext_bb.ll.y, prev_bb_ext.ll.y)) {
1058  this->corners_powerring[n_l] += 2;
1059  }
1060  if (!Math::doubleComp(neighbour_ext_bb.ur.y, prev_bb_ext.ur.y)) {
1061  this->corners_powerring[n_l] += 2;
1062  }
1063  }
1064  // prev bb and neighbour are horizontally intersecting,
1065  // thus the horizontal dimensions were extended
1066  else if (Rect::rectsIntersectHorizontal(neighbour->block->bb, prev_bb)) {
1067 
1068  // check both boundaries separately
1069  if (!Math::doubleComp(neighbour_ext_bb.ll.x, prev_bb_ext.ll.x)) {
1070  this->corners_powerring[n_l] += 2;
1071  }
1072  if (!Math::doubleComp(neighbour_ext_bb.ur.x, prev_bb_ext.ur.x)) {
1073  this->corners_powerring[n_l] += 2;
1074  }
1075  }
1076  }
1077  }
1078 
1079  // calculate cost (amount of intrusion); only required for adapted bbs
1080  //
1081  // note that only _current_ intrusion is considered, i.e. the amount of
1082  // intrusion in any previous merging step is ignored; this is valid since
1083  // the module was already selected as best-cost module, despite any amount
1084  // of intrusion, and the separated bb have been memorized, i.e., the
1085  // starting condition, before considering the neighbour, was a
1086  // non-intruded module
1087  //
1088  cost = intrusion_area / ext_bb.area;
1089  }
1090 
1091  // update cost if required
1092  if (apply_update) {
1093  this->outline_cost = cost;
1094  }
1095 
1096  return cost;
1097 }

1107 inline void MultipleVoltages::CompoundModule::update_power_saving(Block const* block_to_consider) {
1108  unsigned min_voltage_index = this->min_voltage_index();
1109 
1110  // all blocks shall be considered
1111  //
1112  if (block_to_consider == nullptr) {
1113 
1114  // reset previous values
1115  this->power_saving_ = this->power_saving_wasted_ = 0.0;
1116 
1117  for (Block const* b : this->blocks) {
1118 
1119  // for each block, its power saving is given by the theoretical
1120  // max power consumption minus the power consumption achieved
1121  // within this module
1122  this->power_saving_ += (b->power_max() - b->power(min_voltage_index));
1123 
1124  // consider also the ``wasted saving'', that is the difference in
1125  // power saving which is not achievable anymore since each block
1126  // has been assigned to this module which is potentially not the
1127  // best-case / lowest-voltage / lowest-power module
1128  this->power_saving_wasted_ += (b->power(min_voltage_index) - b->power_min());
1129  }
1130  }
1131  // only one specific block shall be considered; update values only, no complete
1132  // recalculation
1133  //
1134  else {
1135  this->power_saving_ += (block_to_consider->power_max() - block_to_consider->power(min_voltage_index));
1136  this->power_saving_wasted_ += (block_to_consider->power(min_voltage_index) - block_to_consider->power_min());
1137  }
1138 }

1145 inline double MultipleVoltages::CompoundModule::cost(double const& max_power_saving, unsigned const& max_corners, MultipleVoltages::Parameters const& parameters) const {
1146  // for the normalization, the min values are fixed: zero for power-saving (for
1147  // trivial modules w/ only highest voltage applicable) and four for corners of
1148  // trivially-shaped(rectangular) modules; add small epsilon to both min values in
1149  // order to avoid division by zero
1150  //
1151  static constexpr double min_corners = 4 + Math::epsilon;
1152  static constexpr double min_power_saving = Math::epsilon;
1153  //
1154  // the max values are derived from all candidate modules; this enables proper
1155  // judgment of quality of any module in terms of weighted sum of cost terms;
1156  // however, this does _not_ allow comparisons between different solutions, i.e.,
1157  // different sets of selected best modules; this will be handled in
1158  // FloorPlanner::evaluateVoltageAssignment where cost are normalized to initial
1159  // values, similar like WL or thermal management cost terms
1160  //
1161 
1162  // this term models the normalized inverse power reduction, with 0 representing
1163  // max power reduction and 1 representing min power reduction, i.e., smaller cost
1164  // represents better solutions
1165  //
1166  double power_saving_term = 1.0 - ((this->power_saving() - min_power_saving) / (max_power_saving - min_power_saving));
1167 
1168  // this term models the normalized number of corners; 0 represents min corners and
1169  // 1 represents max corners, i.e., the less corners the smaller the cost term
1170  //
1171  double corners_term = (static_cast<double>(this->corners_powerring_max()) - min_corners) / (static_cast<double>(max_corners) - min_corners);
1172 
1173  // return weighted sum of terms
1174  //
1175  return (parameters.weight_power_saving * power_saving_term) + (parameters.weight_corners * corners_term);
1176 }
