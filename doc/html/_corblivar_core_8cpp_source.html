33 void CorblivarCore::initCorblivarRandomly(bool const& log, int const& layers, std::vector<Block> const& blocks, bool const& power_aware_assignment) {
34  Direction cur_dir;
35  int die, cur_t, cur_layer;
36  double blocks_area_per_layer, cur_blocks_area;
37  std::vector<Block> blocks_copy;
38  Block const* cur_block;
39 
40  if (log) {
41  std::cout << "Corblivar> ";
42  std::cout << "Initializing Corblivar data for corb on " << layers << " layers; ";
43  if (power_aware_assignment) {
44  std::cout << "w/ power-aware block handling..." << std::endl;
45  }
46  else {
47  std::cout << "w/o power-aware block handling..." << std::endl;
48  }
49  }
50 
51  // local copy; used for blocks order
52  blocks_copy = blocks;
53 
54  // prepare power-aware assignment
55  if (power_aware_assignment) {
56 
57  // init vars
58  blocks_area_per_layer = cur_blocks_area = 0.0;
59  cur_layer = 0;
60 
61  // sort blocks by power density; use local (mutable) copy of blocks
62  sort(blocks_copy.begin(), blocks_copy.end(),
63  // lambda expression
64  [&](Block b1, Block b2) {
65  return b1.power_density() < b2.power_density();
66  }
67  );
68 
69  // determine blocks / die area ratio for balanced assignment
70  blocks_area_per_layer = 0.0;
71  for (Block const& cur_block : blocks) {
72  blocks_area_per_layer += cur_block.bb.area;
73  }
74  blocks_area_per_layer /= layers;
75  }
76 
77  // assign each block to one die, generate L and T as well; consider local, sorted
78  // copy of blocks container
79  for (Block& cur_block_copy : blocks_copy) {
80 
81  // determine related block from original blocks container
82  cur_block = Block::findBlock(cur_block_copy.id, blocks);
83 
84  // for power-aware assignment, fill layers w/ (sorted) blocks until the
85  // dies are evenly occupied
86  if (power_aware_assignment) {
87 
88  cur_blocks_area += cur_block->bb.area;
89 
90  if (cur_blocks_area > blocks_area_per_layer) {
91  cur_layer++;
92  cur_blocks_area = 0.0;
93  }
94 
95  // sanity check to limit die
96  die = std::min(cur_layer, layers - 1);
97  }
98  else {
99  // consider random die
100  die = Math::randI(0, layers);
101  }
102 
103  // memorize layer in block itself
104  cur_block->layer = die;
105 
106  // generate direction L
107  if (Math::randB()) {
108  cur_dir = Direction::HORIZONTAL;
109  }
110  else {
111  cur_dir = Direction::VERTICAL;
112  }
113  // init T-junction to be overlapped as zero, results in initial layout to
114  // be placed ``somewhat diagonally'' into outline
115  cur_t = 0;
116 
117  // store into separate CBL sequences
118  this->dies[die].CBL.S.push_back(std::move(cur_block));
119  this->dies[die].CBL.L.push_back(std::move(cur_dir));
120  this->dies[die].CBL.T.push_back(std::move(cur_t));
121  }
122 
123  if (CorblivarCore::DBG) {
124  for (CorblivarDie const& die : this->dies) {
125  std::cout << "DBG_CORE> ";
126  std::cout << "Init CBL tuples for die " << die.id + 1 << "; " << die.CBL.size() << " tuples:" << std::endl;
127  std::cout << die.CBL.CBLString() << std::endl;
128  std::cout << "DBG_CORE> ";
129  std::cout << std::endl;
130  }
131  }
132 
133  if (log) {
134  std::cout << "Corblivar> ";
135  std::cout << "Done" << std::endl << std::endl;
136  }
137 }

139 bool CorblivarCore::generateLayout(bool const& perform_alignment) {
140  Block const* cur_block;
141  Block const* other_block;
142  std::vector<CorblivarAlignmentReq const*> cur_block_alignment_reqs;
143  CorblivarAlignmentReq const* req_processed;
144 
145  if (CorblivarCore::DBG) {
146  std::cout << "DBG_CORE> ";
147  std::cout << "Performing layout generation..." << std::endl;
148  }
149 
150  // init die pointer
151  this->p = &this->dies[0];
152 
153  // reset die data, i.e., layout generation handler data
154  for (CorblivarDie& die : this->dies) {
155  die.reset();
156  }
157 
158  // reset alignments-in-process list
159  AL.clear();
160 
161  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
162  std::cout << "DBG_ALIGNMENT>" << std::endl;
163  std::cout << "DBG_ALIGNMENT> New layout-generation run..." << std::endl;
164  std::cout << "DBG_ALIGNMENT>" << std::endl;
165  }
166 
167  // perform layout generation in loop (until all blocks are placed)
168  while (true) {
169 
170  // sanity check for empty die
171  if (this->p->getCBL().empty()) {
172  this->p->done = true;
173  }
174 
175  // die done in previous loop; or just marked done since it's empty
176  if (this->p->done) {
177 
178  // continue layout generation on next, yet unfinished die; or
179  // abort loop if all dies marked as done
180  if (!this->switchDie()) {
181  break;
182  }
183  }
184 
185  // dbg logging for current block
186  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
187 
188  std::cout << "DBG_ALIGNMENT> Processing " << this->p->getCBL().tupleString(this->p->pi) << " on die " << this->p->id + 1 << std::endl;
189  }
190 
191  // handle stalled die, i.e., resolve paused alignment process by placing
192  // current block
193  if (this->p->stalled) {
194 
195  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
196  std::cout << "DBG_ALIGNMENT> Resolving stalled die: " << this->p->id + 1;
197  std::cout << " place current block : " << this->p->getCurrentBlock()->id << std::endl;
198  }
199 
200  // place block, increment progress pointer
201  this->p->placeCurrentBlock(perform_alignment);
202  this->p->updateProgressPointerFlag();
203  // mark die as not stalled anymore
204  this->p->stalled = false;
205  }
206  // die is not stalled
207  else {
208  // handle current block on this die
209  cur_block = this->p->getCurrentBlock();
210 
211  // handle block alignment only if desired
212  if (perform_alignment) {
213 
214  // determine related requests for current block
215  cur_block_alignment_reqs = this->findAlignmentReqs(cur_block);
216 
217  // some requests are given, handle them stepwise
218  if (!cur_block_alignment_reqs.empty()) {
219 
220  // handle each request
221  for (auto* cur_req : cur_block_alignment_reqs) {
222 
223  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
224  std::cout << "DBG_ALIGNMENT> Handling alignment request for block " << cur_block->id << std::endl;
225  std::cout << "DBG_ALIGNMENT> Request: " << cur_req->tupleString() << std::endl;
226  }
227 
228  // determine other block of request
229  if (cur_req->s_i->numerical_id == cur_block->numerical_id) {
230  other_block = cur_req->s_j;
231  }
232  else {
233  other_block = cur_req->s_i;
234  }
235 
236  // check if request is already in process; if so,
237  // the die related to the other block is currently
238  // stalled, i.e., waiting for this block to be
239  // placed / both blocks to be aligned
240  req_processed = nullptr;
241  for (auto* req_in_process : this->AL) {
242 
243  if (cur_req->id == req_in_process->id) {
244 
245  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
246  std::cout << "DBG_ALIGNMENT> Request in process; aligning related blocks" << std::endl;
247  }
248 
249  // place/align blocks, but keep
250  // progress pointer for now in
251  // order to handle all alignment
252  // requests for current blocks
253  if (this->alignBlocks(cur_req)) {
254 
255  // in case the request was
256  // handled, memorize
257  // processed request
258  req_processed = cur_req;
259 
260  // mark die related w/ other block
261  // as not stalled any more;
262  // re-enables further layout
263  // generation on that die in next
264  // iterations
265  this->dies[other_block->layer].stalled = false;
266  }
267 
268  break;
269  }
270  }
271 
272  // request is not in process yet;
273  if (req_processed == nullptr) {
274 
275  // stall layout generation on this die;
276  this->dies[cur_block->layer].stalled = true;
277  // memorize alignment as in process;
278  this->AL.push_back(cur_req);
279  // continue layout generation on
280  // die related to other block of
281  // request; only applicable for
282  // regular blocks, i.e., for RBOD,
283  // continue on current die
284  if (other_block->numerical_id != RBOD::NUMERICAL_ID) {
285  this->p = &this->dies[other_block->layer];
286  }
287 
288  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
289  std::cout << "DBG_ALIGNMENT> Request not (yet) in process" << std::endl;
290  std::cout << "DBG_ALIGNMENT> Mark request as in process; stall current die " << cur_block->layer + 1;
291  std::cout << ", continue on die " << this->p->id + 1 << std::endl;
292  }
293  }
294  // request is processed; drop from list of
295  // requests-in-process
296  else {
297  this->AL.remove(req_processed);
298  }
299  }
300 
301  // all requests are handled and further layout generation
302  // to be continued on this die; increment progress pointer
303  // since block and related alignment requests are handled
304  if (!this->dies[cur_block->layer].stalled) {
305 
306  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
307  std::cout << "DBG_ALIGNMENT> All requests handled for block " << cur_block->id << "; continue w/ next block" << std::endl;
308  }
309 
310  this->p->updateProgressPointerFlag();
311  }
312  }
313 
314  // no alignment requested given for current block
315  else {
316  // place block, increment progress pointer
317  this->p->placeCurrentBlock(perform_alignment);
318  this->p->updateProgressPointerFlag();
319  }
320  }
321 
322  // handling block alignment is not desired, simply place blocks
323  else {
324  this->p->placeCurrentBlock(perform_alignment);
325  this->p->updateProgressPointerFlag();
326  }
327  }
328  }
329 
330  if (CorblivarCore::DBG) {
331  std::cout << "DBG_CORE> ";
332  std::cout << "Done" << std::endl;
333  }
334 
335  return true;
336 }

338 bool CorblivarCore::alignBlocks(CorblivarAlignmentReq const* req) {
339  Block const* b1;
340  Block const* b2;
341  CorblivarDie* die_b1;
342  CorblivarDie* die_b2;
343  std::list<Block const*> b1_relev_blocks, b2_relev_blocks;
344  Direction dir_b1, dir_b2;
345  bool b1_shifted, b2_shifted;
346  bool b1_to_shift_horizontal, b1_to_shift_vertical, b2_to_shift_horizontal, b2_to_shift_vertical;
347 
348  // scenario I: both blocks are yet unplaced
349  if (!req->s_i->placed && !req->s_j->placed) {
350 
351  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
352  std::cout << "DBG_ALIGNMENT> Both blocks not placed yet; consider adaptive alignment..." << std::endl;
353  }
354 
355  // local mapping of blocks
356  b1 = req->s_i;
357  b2 = req->s_j;
358 
359  // retrieve related die pointers
360  die_b1 = &this->dies[b1->layer];
361  die_b2 = &this->dies[b2->layer];
362 
363  // pop relevant blocks from related placement stacks
364  b1_relev_blocks = die_b1->popRelevantBlocks();
365  b2_relev_blocks = die_b2->popRelevantBlocks();
366 
367  // first, we need to determine which insertion direction is to be applied
368  // for each block
369  dir_b1 = die_b1->getCurrentDirection();
370  dir_b2 = die_b2->getCurrentDirection();
371 
372  // second, depending on insertion direction, there are various
373  // subscenarios
374  //
375  // subscenario IIa: both blocks have the same insertion direction, i.e.,
376  // can be processed in parallel
377  if (dir_b1 == dir_b2) {
378 
379  // horizontal placement
380  if (dir_b1 == Direction::HORIZONTAL) {
381 
382  // dbg placement direction
383  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
384  std::cout << "DBG_ALIGNMENT> Blocks are handled in parallel, and to be placed horizontally" << std::endl;
385  }
386 
387  // first, determine blocks' y-coordinates
388  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks);
389  die_b2->determCurrentBlockCoords(Coordinate::Y, b2_relev_blocks);
390 
391  // perform shift in y-dir, if required and possible
392  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req);
393  b2_shifted = die_b2->shiftCurrentBlock(Direction::VERTICAL, req);
394 
395  // second, determine block's x-coordinates (depends on
396  // y-coord of relevant blocks and, more generally, on
397  // previously placed, possibly shifted blocks)
398  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks, true);
399  die_b2->determCurrentBlockCoords(Coordinate::X, b2_relev_blocks, true);
400 
401  // perform shift in x-dir, if required and possible
402  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req) || b1_shifted;
403  b2_shifted = die_b2->shiftCurrentBlock(Direction::HORIZONTAL, req) || b2_shifted;
404  }
405  // vertical placement
406  else {
407 
408  // dbg placement direction
409  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
410  std::cout << "DBG_ALIGNMENT> Blocks are handled in parallel, and to be placed vertically" << std::endl;
411  }
412 
413  // first, determine blocks' x-coordinates
414  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks);
415  die_b2->determCurrentBlockCoords(Coordinate::X, b2_relev_blocks);
416 
417  // perform shift in x-dir, if required and possible
418  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req);
419  b2_shifted = die_b2->shiftCurrentBlock(Direction::HORIZONTAL, req);
420 
421  // second, determine block's y-coordinates (depends on
422  // x-coord of relevant blocks and, more generally, on
423  // previously placed, possibly shifted blocks)
424  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks, true);
425  die_b2->determCurrentBlockCoords(Coordinate::Y, b2_relev_blocks, true);
426 
427  // perform shift in y-dir, if required and possible
428  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req) || b1_shifted;
429  b2_shifted = die_b2->shiftCurrentBlock(Direction::VERTICAL, req) || b2_shifted;
430  }
431  }
432  // subscenario IIb: blocks have different insertion direction, i.e.,
433  // cannot be processed in parallel
434  else {
435 
436  // dbg placement direction
437  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
438  std::cout << "DBG_ALIGNMENT> Blocks are to be handled sequentially" << std::endl;
439  }
440 
441  // initially, determine all coordinates
442  if (dir_b1 == Direction::HORIZONTAL) {
443  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks);
444  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks, true);
445  }
446  else {
447  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks);
448  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks, true);
449  }
450  if (dir_b2 == Direction::HORIZONTAL) {
451  die_b2->determCurrentBlockCoords(Coordinate::Y, b2_relev_blocks);
452  die_b2->determCurrentBlockCoords(Coordinate::X, b2_relev_blocks, true);
453  }
454  else {
455  die_b2->determCurrentBlockCoords(Coordinate::X, b2_relev_blocks);
456  die_b2->determCurrentBlockCoords(Coordinate::Y, b2_relev_blocks, true);
457  }
458 
459  // second, determine which block is to be shifted in which direction
460  b1_to_shift_horizontal = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req, true);
461  b1_to_shift_vertical = die_b1->shiftCurrentBlock(Direction::VERTICAL, req, true);
462  b2_to_shift_horizontal = die_b2->shiftCurrentBlock(Direction::HORIZONTAL, req, true);
463  b2_to_shift_vertical = die_b2->shiftCurrentBlock(Direction::VERTICAL, req, true);
464 
465  // third, catch the various cases for possibly required shifting
466  b1_shifted = b2_shifted = false;
467  //
468  // a) one block has to be shifted in both directions
469  if (b1_to_shift_horizontal && b1_to_shift_vertical) {
470 
471  // perform shifting of b1; helper also considers to shift
472  // b2 if required
473  CorblivarCore::sequentialShiftingHelper(
474  die_b1, die_b2, req, b1_relev_blocks, b2_relev_blocks, dir_b1, b1_shifted, b2_shifted);
475  }
476 
477  else if (b2_to_shift_horizontal && b2_to_shift_vertical) {
478 
479  // perform shifting of b2; helper also considers to shift
480  // b1 if required
481  CorblivarCore::sequentialShiftingHelper(
482  die_b2, die_b1, req, b2_relev_blocks, b1_relev_blocks, dir_b2, b2_shifted, b1_shifted);
483  }
484 
485  // b) one or both blocks are to be shifted in their first direction
486  if (!b1_shifted && b1_to_shift_vertical && dir_b1 == Direction::HORIZONTAL) {
487  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req);
488  }
489  else if (!b1_shifted && b1_to_shift_horizontal && dir_b1 == Direction::VERTICAL) {
490  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req);
491  }
492 
493  if (!b2_shifted && b2_to_shift_vertical && dir_b2 == Direction::HORIZONTAL) {
494  b2_shifted = die_b2->shiftCurrentBlock(Direction::VERTICAL, req);
495  }
496  else if (!b2_shifted && b2_to_shift_horizontal && dir_b2 == Direction::VERTICAL) {
497  b2_shifted = die_b2->shiftCurrentBlock(Direction::HORIZONTAL, req);
498  }
499 
500  // c) trivial cases where one or both blocks are to be shifted in
501  // their second direction
502  if (!b1_shifted && b1_to_shift_horizontal && dir_b1 == Direction::HORIZONTAL) {
503  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req);
504  }
505  else if (!b1_shifted && b1_to_shift_vertical && dir_b1 == Direction::VERTICAL) {
506  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req);
507  }
508 
509  if (!b2_shifted && b2_to_shift_horizontal && dir_b2 == Direction::HORIZONTAL) {
510  b2_shifted = die_b2->shiftCurrentBlock(Direction::HORIZONTAL, req);
511  }
512  else if (!b2_shifted && b2_to_shift_vertical && dir_b2 == Direction::VERTICAL) {
513  b2_shifted = die_b2->shiftCurrentBlock(Direction::VERTICAL, req);
514  }
515  }
516 
517  // if a block was shifted, we need to rebuild the related placement stacks
518  // since the corner-block front in both dimensions may be different now
519  //
520  if (b1_shifted) {
521  die_b1->rebuildPlacementStacks(b1_relev_blocks);
522  }
523  // if a block was not shifted, we can simply update the placement stacks
524  else {
525  die_b1->updatePlacementStacks(b1_relev_blocks);
526  }
527 
528  if (b2_shifted) {
529  die_b2->rebuildPlacementStacks(b2_relev_blocks);
530  }
531  else {
532  die_b2->updatePlacementStacks(b2_relev_blocks);
533  }
534 
535  // mark (shifted) blocks as placed
536  b1->placed = true;
537  b2->placed = true;
538 
539  // placement stacks debugging
540  if (CorblivarDie::DBG_STACKS) {
541  die_b1->debugStacks();
542  die_b2->debugStacks();
543  }
544  }
545 
546  // scenario II: both blocks are already placed, further shifting for block
547  // alignment is not feasible
548  else if (req->s_i->placed && req->s_j->placed) {
549 
550  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
551  std::cout << "DBG_ALIGNMENT> Both blocks previously placed; alignment not possible!" << std::endl;
552  }
553  }
554 
555  // scenario III: one block is already placed
556  else {
557 
558  // determine yet unplaced block
559  if (!req->s_i->placed) {
560  b1 = req->s_i;
561  b2 = req->s_j;
562  }
563  else {
564  b1 = req->s_j;
565  b2 = req->s_i;
566  }
567 
568  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
569  std::cout << "DBG_ALIGNMENT> Block " << b2->id << " previously placed; try to shift block " << b1->id << std::endl;
570  }
571 
572  // retrieve related die pointer
573  die_b1 = &this->dies[b1->layer];
574 
575  // sanity check for diff b/w current CBL tuple and current block; that's
576  // happening when the block to be shifted is not the current block, i.e.,
577  // a block to be processed later on; thus, we skip the alignment for now
578  if (b1->numerical_id != die_b1->getCurrentBlock()->numerical_id) {
579 
580  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
581  std::cout << "DBG_ALIGNMENT> Shift block is not current block; abort alignment" << std::endl;
582  }
583 
584  return false;
585  }
586 
587  // pop relevant blocks from related placement stack
588  b1_relev_blocks = die_b1->popRelevantBlocks();
589 
590  // horizontal placement
591  if (die_b1->getCurrentDirection() == Direction::HORIZONTAL) {
592 
593  // dbg placement direction
594  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
595  std::cout << "DBG_ALIGNMENT> Block is to be placed horizontally" << std::endl;
596  }
597 
598  // first, determine block's y-coordinates
599  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks);
600 
601  // perform shift in y-dir, if required and possible
602  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req);
603 
604  // second, determine block's x-coordinates (depends on y-coord of
605  // relevant blocks and, more generally, on previously placed,
606  // possibly shifted blocks)
607  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks, true);
608 
609  // perform shift in x-dir, if required and possible
610  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req) || b1_shifted;
611  }
612  // vertical placement
613  else {
614 
615  // dbg placement direction
616  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
617  std::cout << "DBG_ALIGNMENT> Block is to be placed vertically" << std::endl;
618  }
619 
620  // first, determine block's x-coordinates
621  die_b1->determCurrentBlockCoords(Coordinate::X, b1_relev_blocks);
622 
623  // perform shift in x-dir, if required and possible
624  b1_shifted = die_b1->shiftCurrentBlock(Direction::HORIZONTAL, req);
625 
626  // second, determine block's y-coordinates (depends on x-coord of
627  // relevant blocks and, more generally, on previously placed,
628  // possibly shifted blocks)
629  die_b1->determCurrentBlockCoords(Coordinate::Y, b1_relev_blocks, true);
630 
631  // perform shift in y-dir, if required and possible
632  b1_shifted = die_b1->shiftCurrentBlock(Direction::VERTICAL, req) || b1_shifted;
633  }
634 
635  // if the block was shifted, we need to rebuild the placement stacks since
636  // the corner-block front in both dimensions may be different now
637  if (b1_shifted) {
638  die_b1->rebuildPlacementStacks(b1_relev_blocks);
639  }
640  // if the block was not shifted, we can simply update the placement stacks
641  else {
642  die_b1->updatePlacementStacks(b1_relev_blocks);
643  }
644 
645  // mark shifted block as placed
646  b1->placed = true;
647 
648  // placement stacks debugging
649  if (CorblivarDie::DBG_STACKS) {
650  die_b1->debugStacks();
651  }
652  }
653 
654  return true;
655 }

745 std::vector<CorblivarAlignmentReq const*> CorblivarCore::findAlignmentReqs(Block const* b) const {
746  std::vector<CorblivarAlignmentReq const*> ret;
747 
748  // sanity check for no given requests
749  if (this->A.empty()) {
750  return ret;
751  }
752 
753  // determine requests covering the given block
754  for (CorblivarAlignmentReq const& req : this->A) {
755 
756  if (req.s_i->numerical_id == b->numerical_id || req.s_j->numerical_id == b->numerical_id) {
757 
758  // only consider request which are still in
759  // process, i.e., not both blocks are placed yet
760  if (!req.s_i->placed || !req.s_j->placed) {
761 
762  if (CorblivarAlignmentReq::DBG_LAYOUT_GENERATION) {
763  std::cout << "DBG_ALIGNMENT> Unhandled request: " << req.tupleString() << std::endl;
764  }
765 
766  ret.push_back(&req);
767  }
768  }
769  }
770 
771  // requests w/ placed blocks are considered first; eases handling
772  // of alignment requests such that blocks ready for alignment are
773  // placed/aligned first
774  sort(ret.begin(), ret.end(),
775  // lambda expression
776  [&](CorblivarAlignmentReq const* req1, CorblivarAlignmentReq const* req2) {
777  return (req1->s_i->placed || req1->s_j->placed) && (!req2->s_i->placed && !req2->s_j->placed);
778  }
779  );
780 
781  return ret;
782 }

784 void CorblivarCore::sortCBLs(bool const& log, int const& mode) {
785  std::vector<std::vector<CornerBlockList::Tuple>> tuples;
786  std::vector<CornerBlockList::Tuple> tuples_die;
787  CornerBlockList::Tuple cur_tuple;
788 
789  // log

792

794  std::cout << "Corblivar> ";
795  std::cout << "Sorting CBL tuples by block sizes ..." << std::endl;
796 
797  break;
798  }
799  }
800 
801  // construct temp vectors w/ CBL tuples, assign separate CBL sequences to tuple vectors
802  for (CorblivarDie& die : this->dies) {
803 
804  tuples_die.clear();
805 
806  for (unsigned t = 0; t < die.CBL.size(); t++) {
807 
808  cur_tuple.S = std::move(die.CBL.S[t]);
809  cur_tuple.L = std::move(die.CBL.L[t]);
810  cur_tuple.T = std::move(die.CBL.T[t]);
811 
812  tuples_die.push_back(std::move(cur_tuple));
813  }
814 
815  tuples.push_back(std::move(tuples_die));
816  }
817 
818  // perfom sorting
819  switch (mode) {
820 
821  // sort tuple vector by blocks size, in descending order
822  case CorblivarCore::SORT_CBLS_BY_BLOCKS_SIZE:
823 
824  for (auto& tuples_die : tuples) {
825  sort(tuples_die.begin(), tuples_die.end(),
826  // lambda expression to provide compare function
827  [&](CornerBlockList::Tuple t1, CornerBlockList::Tuple t2) {
828  return t1.S->bb.area > t2.S->bb.area;
829  }
830  );
831  }
832 
833  break;
834  }
835 
836  // reassign CBL sequences from tuple vectors
837  unsigned d = 0;
838  for (CorblivarDie& die : this->dies) {
839 
840  for (unsigned t = 0; t < die.CBL.size(); t++) {
841 
842  die.CBL.S[t] = std::move(tuples[d][t].S);
843  die.CBL.L[t] = std::move(tuples[d][t].L);
844  die.CBL.T[t] = std::move(tuples[d][t].T);
845  }
846 
847  d++;
848  }
849 
850  // log
851  if (log) {
852  std::cout << "Corblivar> ";
853  std::cout << "Done" << std::endl << std::endl;
